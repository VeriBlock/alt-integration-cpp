24 template <typename Element, typename Container>
25 /*iterator*/ auto findDuplicates(BlockIndex<AltBlock>& index,
26  Container& payloads,
27  AltBlockTree& tree)
28  -> decltype(payloads.end()) {
29  const auto startHeight = tree.getParams().getBootstrapBlock().height;
30  // don't look for duplicates in index itself
31  Chain<BlockIndex<AltBlock>> chain(startHeight, index.pprev);
32  std::unordered_set<std::vector<uint8_t>> ids;
33 
34  const auto& storage = tree.getPayloadsIndex();
35  auto duplicates =
36  std::remove_if(payloads.begin(), payloads.end(), [&](const Element& p) {
37  const auto id = getIdVector(p);
38  // ensure existing blocks do not contain this id
39  for (const auto& hash : storage.getContainingAltBlocks(id)) {
40  const auto* candidate = tree.getBlockIndex(hash);
41  if (chain.contains(candidate)) {
42  // duplicate in 'candidate'
43  return true;
44  }
45  }
46 
47  // ensure ids are unique within `pop`
48  bool inserted = ids.insert(id).second;
49  return !inserted;
50  });
51 
52  return duplicates;
53 }
