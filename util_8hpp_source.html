40template <typename BlockTreeT>
41bool loadValidateTree(
42 const BlockTreeT& tree,
43 const std::vector<typename BlockTreeT::stored_index_t>& blocks,
44 ValidationState& state) {
45
46 using block_t = typename BlockTreeT::block_t;
47
48 for (const auto& block : blocks) {
49 if (!validateLoadBlock(tree, block, state)) {
50 return state.Invalid("load-validate-tree",
51 format("Invalid stored {} block {}",
52 block_t::name(),
53 block.toPrettyString()));
54 }
55 }
56 return true;
57}

64template <typename BlockTreeT>
65bool loadTree(BlockTreeT& tree,
66 const typename BlockTreeT::hash_t& tiphash,
67 std::vector<typename BlockTreeT::stored_index_t>& blocks,
68 ValidationState& state) {
69 using stored_index_t = typename BlockTreeT::stored_index_t;
70
71 if (blocks.size() == 0) return true;
72
73 VBK_LOG_WARN("Loading %d %s blocks with tip %s",
74 blocks.size(),
75 BlockTreeT::block_t::name(),
76 HexStr(tiphash));
77 VBK_ASSERT(tree.isBootstrapped() && "tree must be bootstrapped");
78
79 // first, sort them by height
80 std::sort(blocks.begin(),
81 blocks.end(),
82 [](const stored_index_t& a, const stored_index_t& b) {
83 return a.height < b.height;
84 });
85
86 for (const auto& block : blocks) {
87 // load blocks one by one
88 if (!tree.loadBlockForward(block, state)) {
89 return state.Invalid("load-tree");
90 }
91 }
92
93 if (!tree.loadTip(tiphash, state)) {
94 return state.Invalid("load-tree");
95 }
96
97 auto* t = tree.getBestChain().tip();
98 VBK_ASSERT(t != nullptr);
99
100 return true;
101}

108template <typename BlockTreeT>
109void saveTree(
110 BlockTreeT& tree,
111 BlockBatch& batch,
112 std::function<void(const typename BlockTreeT::index_t&)> validator) {
113 using index_t = typename BlockTreeT::index_t;
114 std::vector<const index_t*> dirty_indices;
115
116 // map pair<hash, shared_ptr<index_t>> to vector<index_t*>
117 for (auto& index : tree.getAllBlocks()) {
118 if (index->isDirty()) {
119 index->unsetDirty();
120 dirty_indices.push_back(index);
121 }
122 }
123
124 // sort by height in descending order, because we need to calculate index hash
125 // during saving. this is needed to be progpow-cache friendly, as cache will
126 // be warm for last blocks.
127 std::sort(dirty_indices.begin(),
128 dirty_indices.end(),
129 [](const index_t* a, const index_t* b) {
130 return b->getHeight() < a->getHeight();
131 });
132
133 // write indices
134 for (const index_t* index : dirty_indices) {
135 validator(*index);
136 batch.writeBlock(index->getHash(),
137 tree.makePrevHash(index->getHash()),
138 index->toStoredBlockIndex());
139 }
140
141 batch.writeTip(tree.getBestChain().tip()->getHash());
142}
