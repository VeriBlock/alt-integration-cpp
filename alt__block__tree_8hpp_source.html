478template <typename JsonValue>
479JsonValue ToJSON(const BlockIndex<AltBlock>& i) {
480 auto obj = json::makeEmptyObject<JsonValue>();
481 std::vector<uint256> endorsements;
482 for (const auto& e : i.getContainingEndorsements()) {
483 endorsements.push_back(e.first);
484 }
485 json::putArrayKV(obj, "containingEndorsements", endorsements);
486
487 std::vector<uint256> endorsedBy;
488 for (const auto* e : i.getEndorsedBy()) {
489 endorsedBy.push_back(e->id);
490 }
491 json::putArrayKV(obj, "endorsedBy", endorsedBy);
492 json::putIntKV(obj, "status", i.getStatus());
493
494 auto stored = json::makeEmptyObject<JsonValue>();
495 json::putArrayKV(stored, "vbkblocks", i.getPayloadIds<VbkBlock>());
496 json::putArrayKV(stored, "vtbs", i.getPayloadIds<VTB>());
497 json::putArrayKV(stored, "atvs", i.getPayloadIds<ATV>());
498
499 json::putKV(obj, "stored", stored);
500
501 return obj;
502}
