496 template <typename JsonValue>
497 JsonValue ToJSON(const BlockIndex<AltBlock>& i) {
498  auto obj = json::makeEmptyObject<JsonValue>();
499  std::vector<uint256> endorsements;
500  for (const auto& e : i.getContainingEndorsements()) {
501  endorsements.push_back(e.first);
502  }
503  json::putArrayKV(obj, "containingEndorsements", endorsements);
504 
505  std::vector<uint256> endorsedBy;
506  for (const auto* e : i.getEndorsedBy()) {
507  endorsedBy.push_back(e->id);
508  }
509  json::putArrayKV(obj, "endorsedBy", endorsedBy);
510  json::putIntKV(obj, "status", i.getStatus());
511 
512  auto stored = json::makeEmptyObject<JsonValue>();
513  json::putArrayKV(stored, "vbkblocks", i.getPayloadIds<VbkBlock>());
514  json::putArrayKV(stored, "vtbs", i.getPayloadIds<VTB>());
515  json::putArrayKV(stored, "atvs", i.getPayloadIds<ATV>());
516 
517  json::putKV(obj, "stored", stored);
518 
519  return obj;
520 }
