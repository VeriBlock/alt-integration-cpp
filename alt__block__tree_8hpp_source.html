515 template <typename JsonValue>
516 JsonValue ToJSON(const BlockIndex<AltBlock>& i) {
517  auto obj = json::makeEmptyObject<JsonValue>();
518  std::vector<uint256> endorsements;
519  for (const auto& e : i.getContainingEndorsements()) {
520  endorsements.push_back(e.first);
521  }
522  json::putArrayKV(obj, "containingEndorsements", endorsements);
523 
524  std::vector<uint256> endorsedBy;
525  for (const auto* e : i.getEndorsedBy()) {
526  endorsedBy.push_back(e->id);
527  }
528  json::putArrayKV(obj, "endorsedBy", endorsedBy);
529  json::putIntKV(obj, "status", i.getStatus());
530 
531  auto stored = json::makeEmptyObject<JsonValue>();
532  json::putArrayKV(stored, "vbkblocks", i.getPayloadIds<VbkBlock>());
533  json::putArrayKV(stored, "vtbs", i.getPayloadIds<VTB>());
534  json::putArrayKV(stored, "atvs", i.getPayloadIds<ATV>());
535 
536  json::putKV(obj, "stored", stored);
537 
538  return obj;
539 }
