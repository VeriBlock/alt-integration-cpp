494template <typename JsonValue>
495JsonValue ToJSON(const BlockIndex<AltBlock>& i) {
496 auto obj = json::makeEmptyObject<JsonValue>();
497 std::vector<uint256> endorsements;
498 for (const auto& e : i.getContainingEndorsements()) {
499 endorsements.push_back(e.first);
500 }
501 json::putArrayKV(obj, "containingEndorsements", endorsements);
502
503 std::vector<uint256> endorsedBy;
504 for (const auto* e : i.getEndorsedBy()) {
505 endorsedBy.push_back(e->id);
506 }
507 json::putArrayKV(obj, "endorsedBy", endorsedBy);
508 json::putIntKV(obj, "status", i.getStatus());
509
510 auto stored = json::makeEmptyObject<JsonValue>();
511 json::putArrayKV(stored, "vbkblocks", i.getPayloadIds<VbkBlock>());
512 json::putArrayKV(stored, "vtbs", i.getPayloadIds<VTB>());
513 json::putArrayKV(stored, "atvs", i.getPayloadIds<ATV>());
514
515 json::putKV(obj, "stored", stored);
516
517 return obj;
518}
