30template <typename Block>
31struct BlockIndex : public Block::addon_t {
32 using block_t = Block;
33 using addon_t = typename Block::addon_t;
34 using hash_t = typename block_t::hash_t;
35 using prev_hash_t = typename block_t::prev_hash_t;
36 using height_t = typename block_t::height_t;
37
39 BlockIndex* pprev = nullptr;
40
42 std::set<BlockIndex*> pnext{};
43
45 bool finalized = false;
46
48 BlockIndex(BlockIndex* prev) : pprev(prev) {
49 // FIXME: prev should be a reference
50 VBK_ASSERT(prev != nullptr);
51
52 pprev->pnext.insert(this);
53
54 if (pprev->isFailed()) {
55 setFlag(BLOCK_FAILED_CHILD);
56 }
57
58 height = prev->getHeight() + 1;
59
60 // TODO: what dirtiness should it have here?
61 }
62
64 explicit BlockIndex(height_t _height) : pprev(nullptr), height(_height) {}
65
66 ~BlockIndex() {
67 // revert the block to the "just constructed" state
68 if (!isDeleted()) {
69 deleteTemporarily();
70 }
71
72 if (!isRoot()) {
73 pprev->pnext.erase(this);
74 }
75
76 // FIXME: ideally we want to assert(pnext.empty())
77 // disconnect from next blocks so that next blocks won't have invalid
78 // pointers
79 for (auto* it : pnext) {
80 it->pprev = nullptr;
81 }
82 }
83
84 // BlockIndex is not copyable
85 // BlockIndex is movable
86 // NOLINTNEXTLINE(performance-noexcept-move-constructor)
87 BlockIndex(BlockIndex&& other) = default;
88 // NOLINTNEXTLINE(performance-noexcept-move-constructor)
89 BlockIndex& operator=(BlockIndex&& other) = default;
90
91 bool isDeleted() const { return hasFlags(BLOCK_DELETED); };
92 bool isRoot() const { return pprev == nullptr; };
93
94 void restore() {
95 VBK_ASSERT_MSG(isDeleted(),
96 "tried to restore block %s that is not deleted",
97 toPrettyString());
98
99 // unsetFlag runs setDirty() for us
100 unsetFlag(BLOCK_DELETED);
101 }
102
103 void deleteTemporarily() {
104 VBK_ASSERT_MSG(!isDeleted(),
105 "tried to delete block %s that is already deleted",
106 toPrettyString());
107
108 // FIXME: this is a hack that might eventually bite us
109 addon_t::setNull();
110
111 // preserve FAILED_* flags
112 this->status = (this->status & BLOCK_FAILED_MASK) | BLOCK_VALID_UNKNOWN |
113 BLOCK_DELETED;
114
115 // make it dirty by default
116 setDirty();
117 }
118
119 // loads on-disk fields from 'other' block index.
120 // works like (explicit) copy constructor
121 void mergeFrom(const StoredBlockIndex<Block>& other) {
122 setHeight(other.height);
123 setHeader(other.header);
124 setStatus(other.status);
125
126 other.addon.toInmem(*this);
127 }

144 return this->isValidUpTo(BLOCK_CONNECTED);

147 uint32_t getStatus() const { return status; }
148 void setStatus(uint32_t _status) {
149 if (_status == this->status) return;
150 this->status = _status;
151 setDirty();
152 }

154 uint32_t getValidityLevel() const {
155 auto level = status & BLOCK_VALID_MASK;
156 VBK_ASSERT_MSG(level <= BLOCK_CAN_BE_APPLIED, "unknown validity level");
157 return level;
158 }
159
160 bool isFailed() const { return (status & BLOCK_FAILED_MASK) != 0u; }
161
162 bool isValid(enum BlockStateStatus upTo = BLOCK_VALID_TREE) const {
163 return !isFailed() && isValidUpTo(upTo);
164 }
165
166 bool isValidUpTo(enum BlockStateStatus upTo) const {
167 auto validityLevel = getValidityLevel();
168 VBK_ASSERT_MSG(
169 validityLevel != BLOCK_CAN_BE_APPLIED || !hasFlags(BLOCK_FAILED_POP),
170 "block %s is both BLOCK_CAN_BE_APPLIED and BLOCK_FAILED_POP which are "
171 "mutually exclusive",
172 toPrettyString());
173
174 return getValidityLevel() >= upTo;
175 }
176
177 void setNull() {
178 addon_t::setNull();
179 this->pprev = nullptr;
180 this->pnext.clear();
181 this->height = 0;
182 this->status = BLOCK_VALID_UNKNOWN;
183 // make it dirty by default
184 setDirty();
185 }
186
187 void setNullInmemFields() {
188 addon_t::setNullInmemFields();
189 this->pprev = nullptr;
190 this->pnext.clear();
191 }
192
193 bool raiseValidity(enum BlockStateStatus upTo) {
194 // we can't raise the validity of a block that's known to be invalid due to
195 // PoP it's ok to raise the validity of a block invalidated by altchain
196 if ((status & BLOCK_FAILED_POP) != 0u) {
197 return false;
198 }
199 if ((status & BLOCK_VALID_MASK) < upTo) {
200 VBK_ASSERT_MSG(isRoot() || pprev->getValidityLevel() >= upTo,
201 "attempted to raise the validity level of block %s beyond "
202 "the validity level of its ancestor %s",
203 toPrettyString(),
204 pprev->toPrettyString());
205
206 status = (status & ~BLOCK_VALID_MASK) | upTo;
207 return true;
208 }
209 return false;
210 }
211
212 bool lowerValidity(enum BlockStateStatus upTo) {
213 // we can't lower the validity of a block that's known to be invalid due to
214 // PoP, as that would incorrectly label another validity level as failed.
215 // BLOCK_FAILED_POP has to be cleared first via revalidateSubtree it's ok to
216 // lower the validity of a block invalidated by altchain
217 if ((status & BLOCK_FAILED_POP) != 0u) {
218 return false;
219 }
220 if ((status & BLOCK_VALID_MASK) > upTo) {
221 status = (status & ~BLOCK_VALID_MASK) | upTo;
222 return true;
223 }
224 return false;
225 }

227 void setDirty() { this->dirty = true; }
228 void unsetDirty() { this->dirty = false; }

231 void setFlag(enum BlockValidityStatus s) {
232 auto newStatus = this->status | s;
233 if (newStatus == this->status) return;
234 this->status = newStatus;
235 setDirty();
236 }
237 void unsetFlag(enum BlockValidityStatus s) {
238 auto newStatus = this->status & ~s;
239 if (newStatus == this->status) return;
240 this->status = newStatus;
241 setDirty();
242 }

244 bool hasFlags(enum BlockValidityStatus s) const { return this->status & s; }

345 const BlockIndex* getAncestor(height_t _height) const {
346 VBK_ASSERT(_height >= 0);
347 if (_height > this->height) {
348 return nullptr;
349 }
350
351 // TODO: this algorithm is not optimal. for O(n) seek backwards until we hit
352 // valid height. also it assumes whole blockchain is in memory (pprev is
353 // valid until given height)
354 const BlockIndex* index = this;
355 while (index != nullptr && index->height > _height) {
356 index = index->getPrev();
357 }
358
359 VBK_ASSERT(index == nullptr || index->height == _height);
360 return index;
361 }
362
364 BlockIndex* getAncestor(height_t _height) {
365 const auto& t = as_const(*this);
366 return const_cast<BlockIndex*>(t.getAncestor(_height));
367 }

422template <typename Block>
423const BlockIndex<Block>* getForkBlock(const BlockIndex<Block>& a,
424 const BlockIndex<Block>& b) {
425 const auto initialHeight = std::min(a.getHeight(), b.getHeight());
426
427 for (auto cursorA = a.getAncestor(initialHeight),
428 cursorB = b.getAncestor(initialHeight);
429 cursorA != nullptr && cursorB != nullptr;
430 cursorA = cursorA->getPrev(), cursorB = cursorB->getPrev()) {
431 if (cursorA == cursorB) {
432 return cursorA;
433 }
434 }
435
436 // chain `b` is not connected to `a`
437 return nullptr;
438}

442template <typename Block>
443bool isBlockOutdated(const BlockIndex<Block>& finalBlock,
444 const BlockIndex<Block>& candidate) {
445 // finalBlock is ancestor of candidate
446 if (candidate.getAncestor(finalBlock.getHeight()) == &finalBlock) {
447 return false;
448 }
449
450 // all candidates behind final block are outdated
451 if (candidate.getHeight() < finalBlock.getHeight()) {
452 return true;
453 }
454
455 // all parallel blocks (on same height as final, but not final) are outdated
456 if (candidate.getHeight() == finalBlock.getHeight() &&
457 &finalBlock != &candidate) {
458 return true;
459 }
460
461 // candidate is ancestor of finalBlock and within window
462 if (finalBlock.getAncestor(candidate.getHeight()) == &candidate) {
463 return false;
464 }
465
466 // candidate is on a fork
467 auto* fork = getForkBlock(finalBlock, candidate);
468
469 // candidate does not connect to finalBlock
470 if (fork == nullptr) {
471 return true;
472 }
473
474 // if fork block is outdated, then candidate is also outdated
475 return isBlockOutdated(finalBlock, *fork);
476}
