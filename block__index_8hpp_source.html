192  bool raiseValidity(enum BlockStateStatus upTo) {
193  // we can't raise the validity of a block that's known to be invalid due to
194  // PoP it's ok to raise the validity of a block invalidated by altchain
195  if ((status & BLOCK_FAILED_POP) != 0u) {
196  return false;
197  }
198  if ((status & BLOCK_VALID_MASK) < upTo) {
199  VBK_ASSERT_MSG((pprev == nullptr || pprev->getValidityLevel() >= upTo),
200  "attempted to raise the validity level of block %s beyond "
201  "the validity level of its ancestor %s",
202  toPrettyString(),
203  pprev->toPrettyString());
204 
205  status = (status & ~BLOCK_VALID_MASK) | upTo;
206  return true;
207  }
208  return false;
209  }

211  bool lowerValidity(enum BlockStateStatus upTo) {
212  // we can't lower the validity of a block that's known to be invalid due to
213  // PoP, as that would incorrectly label another validity level as failed.
214  // BLOCK_FAILED_POP has to be cleared first via revalidateSubtree it's ok to
215  // lower the validity of a block invalidated by altchain
216  if ((status & BLOCK_FAILED_POP) != 0u) {
217  return false;
218  }
219  if ((status & BLOCK_VALID_MASK) > upTo) {
220  status = (status & ~BLOCK_VALID_MASK) | upTo;
221  return true;
222  }
223  return false;
224  }

332  const BlockIndex* getAncestor(height_t _height) const {
333  VBK_ASSERT(_height >= 0);
334  if (_height > this->height) {
335  return nullptr;
336  }
337 
338  // TODO: this algorithm is not optimal. for O(n) seek backwards until we hit
339  // valid height. also it assumes whole blockchain is in memory (pprev is
340  // valid until given height)
341  const BlockIndex* index = this;
342  while (index != nullptr && index->height > _height) {
343  index = index->getPrev();
344  }
345 
346  VBK_ASSERT(index == nullptr || index->height == _height);
347  return index;
348  }

430 template <typename Block>
431 const BlockIndex<Block>* getForkBlock(const BlockIndex<Block>& a,
432  const BlockIndex<Block>& b) {
433  const auto initialHeight = std::min(a.getHeight(), b.getHeight());
434 
435  for (auto cursorA = a.getAncestor(initialHeight),
436  cursorB = b.getAncestor(initialHeight);
437  cursorA != nullptr && cursorB != nullptr;
438  cursorA = cursorA->getPrev(), cursorB = cursorB->getPrev()) {
439  if (cursorA == cursorB) {
440  return cursorA;
441  }
442  }
443 
444  // chain `b` is not connected to `a`
445  return nullptr;
446 }

450 template <typename Block>
451 bool isBlockOutdated(const BlockIndex<Block>& finalBlock,
452  const BlockIndex<Block>& candidate) {
453  // finalBlock is ancestor of candidate
454  if (candidate.getAncestor(finalBlock.getHeight()) == &finalBlock) {
455  return false;
456  }
457 
458  // all candidates behind final block are outdated
459  if (candidate.getHeight() < finalBlock.getHeight()) {
460  return true;
461  }
462 
463  // all parallel blocks (on same height as final, but not final) are outdated
464  if (candidate.getHeight() == finalBlock.getHeight() &&
465  &finalBlock != &candidate) {
466  return true;
467  }
468 
469  // candidate is ancestor of finalBlock and within window
470  if (finalBlock.getAncestor(candidate.getHeight()) == &candidate) {
471  return false;
472  }
473 
474  // candidate is on a fork
475  auto* fork = getForkBlock(finalBlock, candidate);
476 
477  // candidate does not connect to finalBlock
478  if (fork == nullptr) {
479  return true;
480  }
481 
482  // if fork block is outdated, then candidate is also outdated
483  return isBlockOutdated(finalBlock, *fork);
484 }

493 template <typename Block>
494 bool DeserializeFromVbkEncoding(
495  ReadStream& stream,
496  BlockIndex<Block>& out,
497  ValidationState& state,
498  typename Block::hash_t precalculatedHash = typename Block::hash_t()) {
499  const auto& name = Block::name();
500  using height_t = typename Block::height_t;
501  if (!stream.readBE<height_t>(out.height, state)) {
502  return state.Invalid(name + "-block-index-height");
503  }
504  Block block{};
505  if (!DeserializeFromRaw(stream, block, state, precalculatedHash)) {
506  return state.Invalid(name + "-block-index-header");
507  }
508  out.setHeader(block);
509  if (!stream.readBE<uint32_t>(out.status, state)) {
510  return state.Invalid(name + "-block-index-status");
511  }
512 
513  using addon_t = typename Block::addon_t;
514  addon_t& addon = out;
515  if (!DeserializeFromVbkEncoding(stream, addon, state)) {
516  return state.Invalid(name + "-block-index-addon");
517  }
518  out.unsetDirty();
519  return true;
520 }
