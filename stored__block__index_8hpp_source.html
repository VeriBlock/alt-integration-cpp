13 template <typename Block>
14 struct StoredBlockIndex {
15  using block_t = Block;
16  using height_t = typename block_t::height_t;
17  using stored_addon_t = typename block_t::stored_addon_t;
18 
19  ~StoredBlockIndex() = default;
20 
21  bool isDeleted() const { return status & BLOCK_DELETED; };
22 
23  void toVbkEncoding(WriteStream& stream) const {
24  using height_t = typename Block::height_t;
25  stream.writeBE<height_t>(height);
26  header->toRaw(stream);
27  stream.writeBE<uint32_t>(status);
28  addon.toVbkEncoding(stream);
29  }
30 
31  std::vector<uint8_t> toVbkEncoding() const {
32  WriteStream stream;
33  toVbkEncoding(stream);
34  return stream.data();
35  }
36 
37  std::string toPrettyString(size_t level = 0) const {
38  return format(
39  "{}{}StoredBlockIndex(height={}, hash={}, status={}, header={}, {})",
40  std::string(level, ' '),
41  Block::name(),
42  height,

44  status,
45  header->toPrettyString(),

47  }

50  height_t height = 0;
51 
53  std::shared_ptr<block_t> header = std::make_shared<block_t>();
54 
56  uint32_t status = BLOCK_VALID_UNKNOWN;
57

61 template <typename Block>
62 bool DeserializeFromVbkEncoding(
63  ReadStream& stream,
64  StoredBlockIndex<Block>& out,
65  ValidationState& state,
66  const AltChainParams& params,
67  typename Block::hash_t precalculatedHash = typename Block::hash_t()) {
68  const auto& name = Block::name();
69  using height_t = typename Block::height_t;
70  if (!stream.readBE<height_t>(out.height, state)) {
71  return state.Invalid(name + "-stored-block-index-height");
72  }
73  Block block{};
74  if (!DeserializeFromRaw(stream, block, state, params, precalculatedHash)) {
75  return state.Invalid(name + "-stored-block-index-header");
76  }
77  out.header = std::make_shared<Block>(block);
78  if (!stream.readBE<uint32_t>(out.status, state)) {
79  return state.Invalid(name + "-stored-block-index-status");
80  }
81 
82  if (!DeserializeFromVbkEncoding(stream, out.addon, state)) {
83  return state.Invalid(name + "-stored-block-index-addon");
84  }
85  return true;
86 }
