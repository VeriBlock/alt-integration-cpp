131template <typename JsonValue>
132JsonValue ToJSON(const BtcBlock& b) {
133 JsonValue object = json::makeEmptyObject<JsonValue>();
134 json::putStringKV(object, "hash", HexStr(b.getHash()));
135 json::putIntKV(object, "version", b.getVersion());
136 json::putStringKV(object, "previousBlock", HexStr(b.getPreviousBlock()));
137 json::putStringKV(object, "merkleRoot", HexStr(b.getMerkleRoot()));
138 json::putIntKV(object, "timestamp", b.getTimestamp());
139 json::putIntKV(object, "bits", b.getDifficulty());
140 json::putIntKV(object, "nonce", b.getNonce());
141 return object;
142}
