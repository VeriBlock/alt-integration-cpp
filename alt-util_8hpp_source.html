25template <typename BlockTreeT>
26std::vector<typename BlockTreeT::block_t> getContext(
27 const BlockTreeT& tree,
28 typename BlockTreeT::hash_t tip,
29 size_t size = std::numeric_limits<size_t>::max()) {
30 std::vector<typename BlockTreeT::block_t> ret;
31 auto* cursor = tree.getBlockIndex(tip);
32 size_t i = 0;
33 while (cursor != nullptr && i++ < size) {
34 ret.push_back(cursor->getHeader());
35 cursor = cursor->pprev;
36 }
37 return ret;
38}

41template <typename BlockTree>
42std::vector<std::vector<uint8_t>> getLastKnownBlocks(const BlockTree& tree,
43 size_t size) {
44 std::vector<std::vector<uint8_t>> ret;
45 ret.reserve(size);
46
47 auto* tip = tree.getBestChain().tip();
48 for (size_t i = 0; i < size && tip != nullptr; i++) {
49 ret.push_back(tip->getHash().asVector());
50 tip = tip->pprev;
51 }
52
53 // reverse them since we add them in reverse order (tip -> genesis)
54 std::reverse(ret.begin(), ret.end());
55 return ret;
56}

59template <typename Block, typename ChainParams>
60bool addBlocks(BlockTree<Block, ChainParams>& tree,
61 const std::vector<std::vector<uint8_t>>& blocks,
62 ValidationState& state) {
63 for (const auto& b : blocks) {
64 Block block = Block::fromRaw(b);
65 if (!tree.acceptBlockHeader(block, state)) {
66 return false;
67 }
68 }
69
70 return true;
71}
