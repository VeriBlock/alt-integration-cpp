43template <typename BlockTreeT>
44std::vector<typename BlockTreeT::block_t> getContext(
45 const BlockTreeT& tree,
46 typename BlockTreeT::hash_t tip,
47 size_t size = std::numeric_limits<size_t>::max()) {
48 std::vector<typename BlockTreeT::block_t> ret;
49 auto* cursor = tree.getBlockIndex(tip);
50 size_t i = 0;
51 while (cursor != nullptr && i++ < size) {
52 ret.push_back(cursor->getHeader());
53 cursor = cursor->pprev;
54 }
55 return ret;
56}

59template <typename BlockTree>
60std::vector<std::vector<uint8_t>> getLastKnownBlocks(const BlockTree& tree,
61 size_t size) {
62 std::vector<std::vector<uint8_t>> ret;
63 ret.reserve(size);
64
65 auto* tip = tree.getBestChain().tip();
66 for (size_t i = 0; i < size && tip != nullptr; i++) {
67 ret.push_back(tip->getHash().asVector());
68 tip = tip->pprev;
69 }
70
71 // reverse them since we add them in reverse order (tip -> genesis)
72 std::reverse(ret.begin(), ret.end());
73 return ret;
74}

77template <typename Block, typename ChainParams>
78bool addBlocks(BlockTree<Block, ChainParams>& tree,
79 const std::vector<std::vector<uint8_t>>& blocks,
80 ValidationState& state) {
81 for (const auto& b : blocks) {
82 Block block = Block::fromRaw(b);
83 if (!tree.acceptBlockHeader(block, state)) {
84 return false;
85 }
86 }
87
88 return true;
89}
