20 struct InmemBlockProvider {
21  template <typename T>
22  using M =
23  std::unordered_map<typename T::hash_t, std::unique_ptr<BlockIndex<T>>>;
24 
25  template <typename T>
26  M<T>& getBlocks();
27 
28  template <typename T>
29  typename T::hash_t getTip() const;
30 
31  template <typename T>
32  void save(const M<T>& m) {
33  getBlocks<T>() = m;
34  }
35 
36  template <typename T>
37  std::vector<std::unique_ptr<BlockIndex<T>>> load() {
38  std::vector<std::unique_ptr<BlockIndex<T>>> ret;
39  auto& m = getBlocks<T>();
40  ret.reserve(m.size());
41 
42  for (auto& b : m) {
43  auto& bi = b.second;
44  auto index = make_unique<BlockIndex<T>>(bi->clone());
45  ret.push_back(std::move(index));
46  }
47 
48  return ret;
49  }

53  M<AltBlock> alt;
