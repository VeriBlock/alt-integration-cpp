88template <typename BlockT>
89struct BlockIteratorImpl : public BlockIterator<BlockT> {
90 ~BlockIteratorImpl() override = default;
91
92 BlockIteratorImpl(std::shared_ptr<StorageIterator> it,
93 const AltChainParams& params)
94 : it_(std::move(it)), params_(params) {}
95
96 void next() override { it_->next(); }
97
98 bool value(StoredBlockIndex<BlockT>& out) const override {
99 std::vector<uint8_t> bytes;
100 if (!it_->value(bytes)) {
101 return false;
102 }
103
104 ValidationState dummy;
105 if (!DeserializeFromVbkEncoding<StoredBlockIndex<BlockT>>(
106 bytes, out, dummy)) {
107 VBK_LOG_ERROR("Can not deserialize StoredBlockIndex<%s>: %s",
108 BlockT::name(),
109 dummy.toString());
110 return false;
111 }
112
113 return true;
114 }
115
116 bool key(typename BlockT::hash_t& out) const override {
117 std::vector<uint8_t> bytes;
118 if (!it_->key(bytes)) {
119 return false;
120 }
121 // remove prefix
122 out = {bytes.begin() + 1, bytes.end()};
123 return true;
124 }
125
126 bool valid() const override {
127 static const uint8_t prefix = block_key<BlockT>({})[0];
128
129 std::vector<uint8_t> key;
130 return it_->valid() && it_->key(key) && !key.empty() && key[0] == prefix;
131 }
132
133 void seek_start() override {
134 static const uint8_t prefix = block_key<BlockT>({})[0];
135
136 it_->seek({prefix});
137 }
138
139 private:
140 std::shared_ptr<StorageIterator> it_;
141 const AltChainParams& params_;
142};

144struct BlockReaderImpl : public BlockReader {
145 ~BlockReaderImpl() override = default;
146
147 BlockReaderImpl(Storage& storage, const AltChainParams& params)
148 : storage_(storage), params_(params) {}
149
150 bool getAltTip(AltBlock::hash_t& out) const override {
151 std::vector<uint8_t> bytes_out;
152 if (!storage_.read(tip_key<AltBlock>(), bytes_out)) {
153 return false;
154 }
155 out = bytes_out;
156 return true;
157 }
158 bool getVbkTip(VbkBlock::hash_t& out) const override {
159 std::vector<uint8_t> bytes_out;
160 if (!storage_.read(tip_key<VbkBlock>(), bytes_out)) {
161 return false;
162 }
163 out = bytes_out;
164 return true;
165 }
166 bool getBtcTip(BtcBlock::hash_t& out) const override {
167 std::vector<uint8_t> bytes_out;
168 if (!storage_.read(tip_key<BtcBlock>(), bytes_out)) {
169 return false;
170 }
171 out = bytes_out;
172 return true;
173 }
174
175 bool getBlock(const AltBlock::prev_hash_t& hash,
176 StoredBlockIndex<AltBlock>& out) const override {
177 return getBlock<AltBlock>(hash, out);
178 }
179
180 bool getBlock(const VbkBlock::prev_hash_t& hash,
181 StoredBlockIndex<VbkBlock>& out) const override {
182 return getBlock<VbkBlock>(hash, out);
183 }
184
185 bool getBlock(const BtcBlock::prev_hash_t& hash,
186 StoredBlockIndex<BtcBlock>& out) const override {
187 return getBlock<BtcBlock>(hash, out);
188 }
189
190 std::shared_ptr<BlockIterator<AltBlock>> getAltBlockIterator()
191 const override {
192 return std::make_shared<BlockIteratorImpl<AltBlock>>(
193 storage_.generateIterator(), params_);
194 }
195 std::shared_ptr<BlockIterator<VbkBlock>> getVbkBlockIterator()
196 const override {
197 return std::make_shared<BlockIteratorImpl<VbkBlock>>(
198 storage_.generateIterator(), params_);
199 }
200 std::shared_ptr<BlockIterator<BtcBlock>> getBtcBlockIterator()
201 const override {
202 return std::make_shared<BlockIteratorImpl<BtcBlock>>(
203 storage_.generateIterator(), params_);
204 }
205
206 private:
207 template <typename block_t>
208 bool getBlock(const typename block_t::prev_hash_t& prev_hash,
209 StoredBlockIndex<block_t>& out) const {
210 std::vector<uint8_t> hash_out;
211 if (!storage_.read(block_prev_key<block_t>(prev_hash), hash_out)) {
212 return false;
213 }
214
215 typename block_t::hash_t hash = hash_out;
216 std::vector<uint8_t> bytes_out;
217 if (!storage_.read(block_key<block_t>(hash), bytes_out)) {
218 return false;
219 }
220
221 ValidationState dummy;
222 if (!DeserializeFromVbkEncoding<StoredBlockIndex<block_t>>(
223 bytes_out, out, dummy)) {
224 VBK_LOG_ERROR("Can not deserialize StoredBlockIndex<%s> block: %s",
225 block_t::name(),
226 dummy.toString());
227 return false;
228 }
229
230 return true;
231 }
232
233 Storage& storage_;
234 const AltChainParams& params_;
235};

257 const BtcBlock::prev_hash_t& prev_hash,

272 batch_.write(tip_key<BtcBlock>(), hash.asVector());
273 }

275 private:
276 WriteBatch& batch_;
277};
