71 struct BlockIteratorImpl : public BlockIterator<BlockT> {
72  ~BlockIteratorImpl() override = default;
73 
74  BlockIteratorImpl(std::shared_ptr<StorageIterator> it,
75  const AltChainParams& params)
76  : it_(it), params_(params) {}
77 
78  void next() override { it_->next(); }
79 
80  bool value(StoredBlockIndex<BlockT>& out) const override {
81  std::vector<uint8_t> bytes;
82  if (!it_->value(bytes)) {
83  return false;
84  }
85  out = AssertDeserializeFromVbkEncoding<StoredBlockIndex<BlockT>>(bytes,
86  params_);
87  return true;
88  }
89 
90  bool key(typename BlockT::hash_t& out) const override {
91  std::vector<uint8_t> bytes;
92  if (!it_->key(bytes)) {
93  return false;
94  }
95  // remove prefix
96  out = {bytes.begin() + 1, bytes.end()};
97  return true;
98  }
99 
100  bool valid() const override {
101  static const uint8_t prefix = block_key<BlockT>({})[0];
102 
103  std::vector<uint8_t> key;
104  return it_->valid() && it_->key(key) && !key.empty() && key[0] == prefix;
105  }
106 
107  void seek_start() override {
108  static const uint8_t prefix = block_key<BlockT>({})[0];
109 
110  it_->seek({prefix});
111  }
112 
113  private:
114  std::shared_ptr<StorageIterator> it_;
115  const AltChainParams& params_;
116 };
