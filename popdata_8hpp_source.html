80template <typename JsonValue, typename T>
81inline void putArrayOfIds(JsonValue& obj,
82 std::string key,
83 const std::vector<T>& t) {
84 JsonValue arr = json::makeEmptyArray<JsonValue>();
85 for (const auto& b : t) {
86 json::arrayPushBack(arr, ToJSON<JsonValue>(b.getId()));
87 }
88
89 json::putKV(obj, key, arr);
90}

92template <typename Value, typename Item>
93void putArrayKV(Value& object,
94 const std::string& key,
95 const std::vector<Item>& val) {
96 auto arr = json::makeEmptyArray<Value>();
97 for (const auto& it : val) {
98 json::arrayPushBack(arr, ToJSON<Value>(it));
99 }
100 json::putKV(object, key, arr);
101}

106template <typename JsonValue>
107JsonValue ToJSON(const PopData& p, bool verbose = false) {
108 JsonValue obj = json::makeEmptyObject<JsonValue>();
109 json::putIntKV(obj, "version", p.version);
110 if (verbose) {
111 detail::putArrayKV(obj, "vbkblocks", p.context);
112 detail::putArrayKV(obj, "vtbs", p.vtbs);
113 detail::putArrayKV(obj, "atvs", p.atvs);
114 } else {
115 detail::putArrayOfIds(obj, "vbkblocks", p.context);
116 detail::putArrayOfIds(obj, "vtbs", p.vtbs);
117 detail::putArrayOfIds(obj, "atvs", p.atvs);
118 }
119
120 return obj;
121}
