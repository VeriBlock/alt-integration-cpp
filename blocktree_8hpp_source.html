29template <typename Block, typename ChainParams>
30struct BlockTree : public BaseBlockTree<Block> {
31 using base = BaseBlockTree<Block>;
32 using block_t = Block;
33 using params_t = ChainParams;

45 const ChainParams& getParams() const { return *param_; }

50 virtual void bootstrapWithGenesis(const block_t& block) {
51 VBK_ASSERT(!base::isBootstrapped() && "already bootstrapped");
52 ValidationState state;
53 this->bootstrap(0, block, state);
54 }

65 virtual void bootstrapWithChain(height_t startHeight,
66 const std::vector<block_t>& chain) {
67 VBK_ASSERT(!base::isBootstrapped() && "already bootstrapped");
68 ValidationState state;
69 VBK_ASSERT_MSG(!chain.empty(), "bootstrap chain must not be empty");
70 VBK_ASSERT_MSG(
71 chain.size() >= param_->numBlocksForBootstrap(),
72 "bootstrap chain is too small (%d) expected at least %d blocks",
73 chain.size(),
74 param_->numBlocksForBootstrap());
75
76 // pick first block from the chain, bootstrap with a single block
77 auto genesis = chain[0];
78 this->bootstrap(startHeight, genesis, state);
79
80 // apply the rest of the blocks from the chain on top of our bootstrap
81 // block. disable difficulty checks, because we have not enough blocks in
82 // our store (yet) to check it correctly
83 for (size_t i = 1, size = chain.size(); i < size; i++) {
84 auto& block = chain[i];
85 bool ok = this->acceptBlockHeaderImpl(
86 std::make_shared<block_t>(block), state, false);
87 VBK_ASSERT_MSG(ok,
88 "found statelssly invalid bootstrap block: %d, %s",
89 i,
90 state.toString());
91 auto* index = base::getBlockIndex(block.getHash());
92 VBK_ASSERT(index);
93 index->setIsBootstrap(true);
94 }
95 }
96
97 bool acceptBlockHeader(const block_t& block, ValidationState& state) {
98 return acceptBlockHeaderImpl(std::make_shared<block_t>(block), state, true);
99 }

101 bool acceptBlockHeader(const std::shared_ptr<block_t>& block,

117 bool loadBlockForward(const stored_index_t& index,
118 ValidationState& state) override {
119 if (!checkBlock(*index.header, state, *param_)) {
120 return state.Invalid("bad-header");
121 }
122
123 const auto hash = index.header->getHash();
124 if (!base::loadBlockForward(index, state)) {
125 return false;
126 }
127
128 auto* current = base::getBlockIndex(hash);
129 VBK_ASSERT(current);
130
131 // we only check blocks contextually if they are not bootstrap blocks, and
132 // previous block exists
133 if (!current->isRoot() && !current->hasFlags(BLOCK_BOOTSTRAP) &&
134 !contextuallyCheckBlock(
135 *current->pprev, current->getHeader(), state, *param_)) {
136 return state.Invalid("bad-block-contextually");
137 }
138
139 // recover chainwork
140 current->chainWork = getBlockProof(current->getHeader());
141 if (!current->isRoot()) {
142 current->chainWork += current->pprev->chainWork;
143 }
144
145 // clear blockOfProofEndorsements inmem field
146 current->clearBlockOfProofEndorsement();
147
148 current->raiseValidity(BLOCK_VALID_TREE);
149 return true;
150 }
151
154 bool isBlockOld(height_t height) const {
155 auto* tip = base::getBestChain().tip();
156 VBK_ASSERT(tip);
157
158 return tip->getHeight() - height > getParams().getOldBlocksWindow();
159 }
160
162 bool isBlockOld(const hash_t& hash) const {
163 auto* index = base::getBlockIndex(hash);
164 if (index == nullptr) {
165 // block is unknown, so not "old"
166 return false;
167 }
168
169 return isBlockOld(index->getHeight());
170 }

172 void finalizeBlocks() {
173 auto* tip = this->getBestChain().tip();
174 VBK_ASSERT(tip && "BTC tree must be bootstrapped");
175
176 int32_t firstBlockHeight =
177 tip->getHeight() - this->getParams().getOldBlocksWindow();
178 int32_t bootstrapBlockHeight = this->getRoot().getHeight();
179 firstBlockHeight = std::max(bootstrapBlockHeight, firstBlockHeight);
180 auto* finalizedIndex = this->getBestChain()[firstBlockHeight];
181 VBK_ASSERT_MSG(finalizedIndex != nullptr, "Invalid BTC tree state");
182
183 this->finalizeBlockImpl(*finalizedIndex,
184 this->getParams().preserveBlocksBehindFinal());
185 }

187 protected:
188 const ChainParams* param_ = nullptr;
189
190 bool acceptBlockHeaderImpl(const std::shared_ptr<block_t>& block,
191 ValidationState& state,
192 bool shouldContextuallyCheck) {
193 index_t* index = nullptr;
194 if (!validateAndAddBlock(block, state, shouldContextuallyCheck, &index)) {
195 return false;
196 }
197
198 VBK_ASSERT(index);
199 if (!shouldContextuallyCheck) {
200 // this is a bootstrap block
201 index->setFlag(BLOCK_BOOTSTRAP);
202 }
203
204 base::tryAddTip(index);
205
206 // don't defer fork resolution in the acceptBlockHeader+addPayloads flow
207 // until the validation hole is plugged
208 determineBestChain(*index, state);
209
210 return true;
211 }

213 void bootstrap(height_t height,
214 const block_t& block,
215 ValidationState& state) {
216 bootstrap(height, std::make_shared<block_t>(block), state);
217 }
218
219 void bootstrap(height_t height,
220 std::shared_ptr<block_t> block,
221 ValidationState& state) {
222 bool ok = checkBlock(*block, state, *param_);
223 VBK_ASSERT_MSG(ok,
224 "found statelessly invalid bootstrap block in %s tree: %s",
225 block_t::name(),
226 state.toString());
227
228 auto* index = base::insertBlockHeader(block, height);
229 VBK_ASSERT(index != nullptr &&
230 "insertBlockHeader should have never returned nullptr");
231
232 index->setHeight(height);
233 base::activeChain_ = Chain<index_t>(height, index);
234 index->setIsBootstrap(true);
235
236 VBK_ASSERT(index->finalized);
237 VBK_ASSERT(base::isBootstrapped());
238 VBK_ASSERT(base::getBlockIndex(index->getHash()) != nullptr &&
239 "getBlockIndex must be able to find the block added by "
240 "insertBlockHeader");
241
242 base::tryAddTip(index);
243 index->setFlag(BLOCK_ACTIVE);
244 bool success = index->raiseValidity(BLOCK_CAN_BE_APPLIED);
245 VBK_ASSERT(success);
246 index->setFlag(BLOCK_BOOTSTRAP);
247 base::appliedBlockCount = 1;
248 }
249
250 bool validateAndAddBlock(const std::shared_ptr<block_t>& block,
251 ValidationState& state,
252 bool shouldContextuallyCheck,
253 index_t** ret) {
254 if (!checkBlock(*block, state, *param_)) {
255 return state.Invalid(block_t::name() + "-check-block");
256 }
257
258 // we must know previous block
259 auto* prev = base::getBlockIndex(block->getPreviousBlock());
260 if (prev == nullptr) {
261 return state.Invalid(
262 block_t::name() + "-bad-prev-block",
263 "can not find previous block: " + HexStr(block->getPreviousBlock()));
264 }
265
266 if (shouldContextuallyCheck &&
267 !contextuallyCheckBlock(*prev, *block, state, *param_)) {
268 return state.Invalid(block_t::name() + "-contextually-check-block");
269 }
270
271 auto index = base::insertBlockHeader(block);
272 VBK_ASSERT(index != nullptr &&
273 "insertBlockHeader should have never returned nullptr");
274
275 if (ret) {
276 *ret = index;
277 }
278
279 index->raiseValidity(BLOCK_CONNECTED);
280
281 // if prev block is invalid, mark this block as invalid
282 if (!prev->isValid()) {
283 index->setFlag(BLOCK_FAILED_CHILD);
284 return state.Invalid(
285 block_t::name() + "-bad-chain",
286 format("Previous block is invalid={}", prev->toPrettyString()));
287 }
288
289 return true;
290 }
291
292 void determineBestChain(index_t& candidate, ValidationState& state) override {
293 auto bestTip = base::getBestChain().tip();
294 VBK_ASSERT(bestTip != nullptr && "must be bootstrapped");
295
296 if (bestTip == &candidate) {
297 return;
298 }
299
300 // do not even consider invalid candidates
301 if (!candidate.isValid()) {
302 return;
303 }
304
305 if (bestTip->chainWork < candidate.chainWork) {
307 this->setState(candidate, state);
308 }
309 }
310
312 void onBlockInserted(index_t* newIndex) override final {
313 newIndex->chainWork = getBlockProof(newIndex->getHeader());
314 if (!newIndex->isRoot()) {
315 newIndex->chainWork += newIndex->pprev->chainWork;
316 }
317 }
318};
