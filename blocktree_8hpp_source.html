70  virtual bool bootstrapWithChain(height_t startHeight,
71  const std::vector<block_t>& chain,
72  ValidationState& state) {
73  VBK_ASSERT(!base::isBootstrapped() && "already bootstrapped");
74  if (chain.empty()) {
75  return state.Invalid(block_t::name() + "-bootstrap-empty-chain",
76  "provided bootstrap chain is empty");
77  }
78 
79  if (chain.size() < param_->numBlocksForBootstrap()) {
80  return state.Invalid(
81  block_t::name() + "-bootstrap-small-chain",
82  "number of blocks in the provided chain is too small: " +
83  std::to_string(chain.size()) + ", expected at least " +
84  std::to_string(param_->numBlocksForBootstrap()));
85  }
86 
87  // pick first block from the chain, bootstrap with a single block
88  auto genesis = chain[0];
89  if (!this->bootstrap(startHeight, genesis, state)) {
90  return state.Invalid(block_t::name() + "-blocktree-bootstrap");
91  }
92 
93  // apply the rest of the blocks from the chain on top of our bootstrap
94  // block. disable difficulty checks, because we have not enough blocks in
95  // our store (yet) to check it correctly
96  for (size_t i = 1, size = chain.size(); i < size; i++) {
97  auto& block = chain[i];
98  if (!this->acceptBlockHeaderImpl(
99  std::make_shared<block_t>(block), state, false)) {
100  return state.Invalid(block_t::name() + "-blocktree-accept");
101  }
102  auto* index = base::getBlockIndex(block.getHash());
103  VBK_ASSERT(index);
104  index->setIsBootstrap(true);
105  }
106 
107  return true;
108  }
109 
110  bool acceptBlockHeader(const block_t& block, ValidationState& state) {
111  return acceptBlockHeaderImpl(std::make_shared<block_t>(block), state, true);
112  }

130  bool loadBlockForward(const stored_index_t& index,
131  ValidationState& state) override {
132  if (!checkBlock(*index.header, state, *param_)) {
133  return state.Invalid("bad-header");
134  }
135 
136  const auto hash = index.header->getHash();
137  if (!base::loadBlockForward(index, state)) {
138  return false;
139  }
140 
141  auto* current = base::getBlockIndex(hash);
142  VBK_ASSERT(current);
143 
144  // we only check blocks contextually if they are not bootstrap blocks, and
145  // previous block exists
146  if (!current->isRoot() && !current->hasFlags(BLOCK_BOOTSTRAP) &&
147  !contextuallyCheckBlock(
148  *current->pprev, current->getHeader(), state, *param_)) {
149  return state.Invalid("bad-block-contextually");
150  }
151 
152  // recover chainwork
153  current->chainWork = getBlockProof(current->getHeader());
154  if (!current->isRoot()) {
155  current->chainWork += current->pprev->chainWork;
156  }
157 
158  // clear blockOfProofEndorsements inmem field
159  current->clearBlockOfProofEndorsement();
160 
161  current->raiseValidity(BLOCK_VALID_TREE);
162  return true;
163  }

197  bool acceptBlockHeaderImpl(const std::shared_ptr<block_t>& block,
198  ValidationState& state,
199  bool shouldContextuallyCheck) {
200  index_t* index = nullptr;
201  if (!validateAndAddBlock(block, state, shouldContextuallyCheck, &index)) {
202  return false;
203  }
204 
205  VBK_ASSERT(index);
206  if (!shouldContextuallyCheck) {
207  // this is a bootstrap block
208  index->setFlag(BLOCK_BOOTSTRAP);
209  }
210 
211  base::tryAddTip(index);
212 
213  // don't defer fork resolution in the acceptBlockHeader+addPayloads flow
214  // until the validation hole is plugged
215  determineBestChain(*index, state);
216 
217  return true;
218  }

226  bool bootstrap(height_t height,
227  std::shared_ptr<block_t> block,
228  ValidationState& state) {
229  if (!checkBlock(*block, state, *param_)) {
230  return state.Invalid(block_t::name() + "-bootstrap");
231  }
232 
233  auto* index = base::insertBlockHeader(block, height);
234  VBK_ASSERT(index != nullptr &&
235  "insertBlockHeader should have never returned nullptr");
236 
237  index->setHeight(height);
238  base::activeChain_ = Chain<index_t>(height, index);
239  index->setIsBootstrap(true);
240 
241  VBK_ASSERT(base::isBootstrapped());
242  VBK_ASSERT(base::getBlockIndex(index->getHash()) != nullptr &&
243  "getBlockIndex must be able to find the block added by "
244  "insertBlockHeader");
245 
246  base::tryAddTip(index);
247  index->setFlag(BLOCK_ACTIVE);
248  bool success = index->raiseValidity(BLOCK_CAN_BE_APPLIED);
249  VBK_ASSERT(success);
250  index->setFlag(BLOCK_BOOTSTRAP);
251  base::appliedBlockCount = 1;
252  return true;
253  }
254 
255  bool validateAndAddBlock(const std::shared_ptr<block_t>& block,
256  ValidationState& state,
257  bool shouldContextuallyCheck,
258  index_t** ret) {
259  if (!checkBlock(*block, state, *param_)) {
260  return state.Invalid(block_t::name() + "-check-block");
261  }
262 
263  // we must know previous block
264  auto* prev = base::getBlockIndex(block->getPreviousBlock());
265  if (prev == nullptr) {
266  return state.Invalid(
267  block_t::name() + "-bad-prev-block",
268  "can not find previous block: " + HexStr(block->getPreviousBlock()));
269  }
270 
271  if (shouldContextuallyCheck &&
272  !contextuallyCheckBlock(*prev, *block, state, *param_)) {
273  return state.Invalid(block_t::name() + "-contextually-check-block");
274  }
275 
276  auto index = base::insertBlockHeader(block);
277  VBK_ASSERT(index != nullptr &&
278  "insertBlockHeader should have never returned nullptr");
279 
280  if (ret) {
281  *ret = index;
282  }
283 
284  index->raiseValidity(BLOCK_CONNECTED);
285 
286  // if prev block is invalid, mark this block as invalid
287  if (!prev->isValid()) {
288  index->setFlag(BLOCK_FAILED_CHILD);
289  return state.Invalid(
290  block_t::name() + "-bad-chain",
291  format("Previous block is invalid={}", prev->toPrettyString()));
292  }
293 
294  return true;
295  }
296 
297  void determineBestChain(index_t& candidate, ValidationState& state) override {
298  auto bestTip = base::getBestChain().tip();
299  VBK_ASSERT(bestTip != nullptr && "must be bootstrapped");
300 
301  if (bestTip == &candidate) {
302  return;
303  }
304 
305  // do not even consider invalid candidates
306  if (!candidate.isValid()) {
307  return;
308  }
309 
310  if (bestTip->chainWork < candidate.chainWork) {
312  this->setState(candidate, state);
313  }
314  }
315 
317  void onBlockInserted(index_t* newIndex) override final {
318  newIndex->chainWork = getBlockProof(newIndex->getHeader());
319  if (!newIndex->isRoot()) {
320  newIndex->chainWork += newIndex->pprev->chainWork;
321  }
322  }
323 };
