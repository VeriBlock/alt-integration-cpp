77template <typename JsonValue>
78JsonValue ToJSON(const VbkPopTx& tx) {
79 JsonValue obj = json::makeEmptyObject<JsonValue>();
80 tx.networkOrType.networkType.template putJson<JsonValue>(obj);
81 json::putStringKV(obj, "hash", tx.getHash().toHex());
82 json::putIntKV(obj, "type", tx.networkOrType.typeId);
83 json::putStringKV(obj, "address", tx.address.toString());
84 json::putKV(obj, "publishedBlock", ToJSON<JsonValue>(tx.publishedBlock));
85 json::putStringKV(
86 obj, "bitcoinTransaction", SerializeToRawHex(tx.bitcoinTransaction));
87 json::putKV(obj, "merklePath", ToJSON<JsonValue>(tx.merklePath));
88 json::putKV(obj, "blockOfProof", ToJSON<JsonValue>(tx.blockOfProof));
89 json::putArrayKV(obj, "blockOfProofContext", tx.blockOfProofContext);
90 json::putStringKV(obj, "signature", HexStr(tx.signature));
91 json::putStringKV(obj, "publicKey", HexStr(tx.publicKey));
92 return obj;
93}
