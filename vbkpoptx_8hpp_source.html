88template <typename JsonValue>
89JsonValue ToJSON(const VbkPopTx& tx) {
90 JsonValue obj = json::makeEmptyObject<JsonValue>();
91 tx.networkOrType.networkType.template putJson<JsonValue>(obj);
92 json::putStringKV(obj, "hash", tx.getHash().toHex());
93 json::putIntKV(obj, "type", tx.networkOrType.typeId);
94 json::putStringKV(obj, "address", tx.address.toString());
95 json::putKV(obj, "publishedBlock", ToJSON<JsonValue>(tx.publishedBlock));
96 json::putStringKV(
97 obj, "bitcoinTransaction", SerializeToRawHex(tx.bitcoinTransaction));
98 json::putKV(obj, "merklePath", ToJSON<JsonValue>(tx.merklePath));
99 json::putKV(obj, "blockOfProof", ToJSON<JsonValue>(tx.blockOfProof));
100 json::putArrayKV(obj, "blockOfProofContext", tx.blockOfProofContext);
101 json::putStringKV(obj, "signature", HexStr(tx.signature));
102 json::putStringKV(obj, "publicKey", HexStr(tx.publicKey));
103 return obj;
104}
