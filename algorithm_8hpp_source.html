21 template <typename A, typename B>

98 template <typename T>
99 bool erase_last_item_if(std::vector<T>& v,
100  const std::function<bool(const T&)>& locator) {
101  // find and erase the last occurrence of item
102  size_t last = v.size();
103  for (size_t i = v.size(); i-- > 0;) {
104  const T& el = v[i];
105  if (locator(el)) {
106  last = i;
107  break;
108  }
109  }
110 
111  if (last >= v.size()) {
112  // not found
113  return false;
114  }
115 
116  auto it = v.begin() + last;
117  v.erase(it);
118 
119  return true;
120 }

124 template <typename container_t, typename val_t>
125 void erase_if(container_t& c, std::function<bool(const val_t&)> pred) {
126  for (auto it = c.begin(); it != c.end();) {
127  it = pred(*it) ? c.erase(it) : std::next(it);
