6 #ifndef ALT_INTEGRATION_INCLUDE_VERIBLOCK_RING_BUFFER_HPP_
7 #define ALT_INTEGRATION_INCLUDE_VERIBLOCK_RING_BUFFER_HPP_
8 
9 #include <vector>
10 
11 namespace altintegration {
12 
14 template <class T>
15 class ring_buffer {
16  public:
17  using value_type = T;
18  using reference = T &;
19  using const_reference = const T &;
20  using size_type = size_t;
21  using circularBuffer = std::vector<value_type>;
22 
23  ring_buffer(size_type size)
24  : m_array(size + 1), m_array_size(size + 1), m_head(0), m_tail(0) {}
25 
26  reference front() {
27  assert(!empty());
28  return m_array[m_head];
29  }
30  const_reference front() const {
31  assert(!empty());
32  return m_array[m_head];
33  }
34 
35  reference back() {
36  assert(!empty());
37  return m_array[prev(m_tail)];
38  }
39  const_reference back() const {
40  assert(!empty());
41  return m_array[prev(m_tail)];
42  }
43 
44  size_type capacity() const {
45  assert(m_array_size > 0);
46  return m_array_size - 1;
47  }
48 
49  size_type size() const {
50  return (m_tail + m_array_size - m_head) % m_array_size;
51  }
52 
53  bool empty() const { return m_tail == m_head; }
54  bool full() const { return m_head == next(m_tail); }
55 
56  const_reference operator[](size_type index) const {
57  return m_array[(m_head + index) % m_array_size];
58  }
59 
60  void push_back(const value_type &item) {
61  m_array[m_tail] = item;
62  increment_tail();
63  if (m_head == m_tail) increment_head();
64  }
65 
66  value_type pop_front() {
67  assert(!empty());
68  reference item = front();
69  increment_head();
70  return item;
71  }
72 
73  value_type pop_back() {
74  assert(!empty());
75  reference item = back();
76  m_tail = prev(m_tail);
77  return item;
78  }
79 
80  void clear() {
81  m_head = 0;
82  m_tail = 0;
83  }
84 
85  private:
86  circularBuffer m_array;
87  const size_type m_array_size;
88  size_type m_head;
89  size_type m_tail;
90 
91  size_type next(size_type cur) const { return (cur + 1) % m_array_size; }
92  size_type prev(size_type cur) const {
93  return (cur + m_array_size - 1) % m_array_size;
94  }
95 
96  void increment_head() { m_head = next(m_head); }
97  void increment_tail() { m_tail = next(m_tail); }
98 };
99 
100 } // namespace altintegration
