73 template <typename ProtectingBlockT, typename ProtectingChainParams>
74 KeystoneContext getKeystoneContext(
75  const ProtoKeystoneContext<ProtectingBlockT>& pkc,
76  const BlockTree<ProtectingBlockT, ProtectingChainParams>& tree) {
77  VBK_TRACE_ZONE_SCOPED;
78  int earliestEndorsementIndex = NO_ENDORSEMENT;
79  for (const auto* btcIndex : pkc.referencedByBlocks) {
80  if (btcIndex == nullptr) {
81  continue;
82  }
83 
84  auto endorsementIndex = btcIndex->getHeight();
85  if (endorsementIndex >= earliestEndorsementIndex) {
86  continue;
87  }
88 
89  bool EnableTimeAdjustment = tree.getParams().EnableTimeAdjustment();
90  if (!EnableTimeAdjustment ||
91  pkc.timestampOfEndorsedBlock < btcIndex->getTimestamp()) {
92  earliestEndorsementIndex = endorsementIndex;
93  continue;
94  }
95 
96  // look at the future BTC blocks and set the
97  // earliestEndorsementIndex to a future Bitcoin block
98  auto best = tree.getBestChain();
99  for (int adjustedEndorsementIndex = endorsementIndex + 1;
100  adjustedEndorsementIndex <= best.chainHeight();
101  adjustedEndorsementIndex++) {
102  // Ensure that the keystone's block time isn't later than the
103  // block time of the Bitcoin block it's endorsed in
104  auto* index = best[adjustedEndorsementIndex];
105  VBK_ASSERT(index != nullptr);
106  if (pkc.timestampOfEndorsedBlock < index->getTimestamp()) {
107  // Timestamp of VeriBlock block is lower than Bitcoin block,
108  // set this as the adjusted index if another lower index has
109  // not already been set
110  if (adjustedEndorsementIndex < earliestEndorsementIndex) {
111  earliestEndorsementIndex = adjustedEndorsementIndex;
112  }
113 
114  // Always break; we found a valid Bitcoin index and any
115  // future adjustedEndorsementIndex is going to be higher
116  break;
117  } // end if
118  } // end for
119  } // end for
120 
121  return {pkc.blockHeight, earliestEndorsementIndex};
122 }

124 template <typename ProtectedBlockT,
125  typename ProtectingBlockT,
126  typename ProtectingChainParams,
127  typename ProtectedChainParams>
128 ProtoKeystoneContext<ProtectingBlockT> getProtoKeystoneContext(
129  int keystoneToConsider,
130  const ChainSlice<BlockIndex<ProtectedBlockT>>& chain,
131  const std::unordered_set<typename ProtectedBlockT::hash_t>&
132  allHashesInChain,
133  const BlockTree<ProtectingBlockT, ProtectingChainParams>& tree,
134  const ProtectedChainParams& config) {
135  VBK_TRACE_ZONE_SCOPED;
136  auto ki = config.getKeystoneInterval();
137  auto* tip = chain.tip();
138  VBK_ASSERT(tip != nullptr && "chain must not be empty");
139 
140  auto highestConnectingBlock =
141  highestBlockWhichConnectsKeystoneToPrevious(keystoneToConsider, ki);
142  auto highestPossibleEndorsedBlockHeaderHeight = tip->getHeight();
143  auto highestEndorsedBlock = std::min(
144  highestConnectingBlock, highestPossibleEndorsedBlockHeaderHeight);
145 
146  ProtoKeystoneContext<ProtectingBlockT> pkc(
147  keystoneToConsider, chain[keystoneToConsider]->getTimestamp());
148 
149  // Find the endorsements of the keystone block and other blocks which
150  // reference it, and look at the earliest Bitcoin block that any of those
151  // endorsements are contained within.
152  for (auto relevantEndorsedBlock = keystoneToConsider;
153  relevantEndorsedBlock <= highestEndorsedBlock;
154  relevantEndorsedBlock++) {
155  auto* index = chain[relevantEndorsedBlock];
156 
157  // chain must contain relevantEndorsedBlock
158  VBK_ASSERT(index != nullptr);
159 
160  for (const auto* e : index->getEndorsedBy()) {
161  if (!allHashesInChain.count(e->containingHash)) {
162  // do not count endorsement whose containingHash is not on the same
163  // chain as 'endorsedHash'
164  continue;
165  }
166 
167  auto* ind = tree.getBlockIndex(e->blockOfProof);
168  VBK_ASSERT(ind != nullptr &&
169  "state corruption: could not find the block of proof of "
170  "an applied endorsement");
171  if (!tree.getBestChain().contains(ind)) {
172  continue;
173  }
174 
175  // include only endorsements that are on best chain of protecting chain,
176  // and whose 'containingHash' is on the same chain as 'endorsedHash'
177  pkc.referencedByBlocks.insert(ind);
178 
179  } // end for
180  } // end for
181 
182  return pkc;
183 }

270 template <typename PublicationView>
271 int comparePopScoreImpl(PublicationView& a, PublicationView& b) {
272  VBK_TRACE_ZONE_SCOPED;
273 
274  if (a.empty() && b.empty()) {
275  return 0;
276  }
277 
278  if (a.empty()) {
279  // a is empty, b is not
280  return -1;
281  }
282 
283  if (b.empty()) {
284  // b is empty, a is not
285  return 1;
286  }
287 
288  VBK_LOG_DEBUG(
289  "Comparing POP scores of chains A(first=%d, tip=%d) "
290  "and B(first=%d, tip=%d)",
291  a.firstKeystone(),
292  a.lastKeystone(),
293  b.firstKeystone(),
294  b.lastKeystone());
295 
296  int earliestKeystone = a.firstKeystone();
297  VBK_ASSERT(earliestKeystone == b.firstKeystone());
298  int latestKeystone = (std::max)(a.lastKeystone(), b.lastKeystone());
299 
300  auto& config = a.getConfig();
301  VBK_ASSERT(&config == &b.getConfig()); // FIXME: comparing pointers is UGLY
302  // clang-format off
303  // If either chain has a keystone the other chain is missing, the other chain
304  // goes outside finality.
305  //
306  // Example:
307  // Chain A has keystones VBK20:BTC100, VBK40:BTC101, VBK60:BTC103, VBK100:BTC104
308  // Chain B has keystones VBK20:BTC100, VBK40:BTC101, VBK100:BTC102
309  //
310  // Chain A has keystone 60 but chain B does not, so Chain B is
311  // chopped to VBK20:BTC100, VBK40:BTC101.
312 
313  // Also, the chain goes out of finality if it contains a keystone which violates
314  // the Bitcoin finality delay based on the previous keystone in the chain.
315  // clang-format on
316 
317  bool aOutsideFinality = false;
318  bool bOutsideFinality = false;
319  int chainAscore = 0;
320  int chainBscore = 0;
321  int previousPublicationA = NO_ENDORSEMENT;
322  int previousPublicationB = NO_ENDORSEMENT;
323 
324  for (int keystoneToCompare = earliestKeystone;
325  keystoneToCompare <= latestKeystone;
326  keystoneToCompare = a.nextKeystoneAfter(keystoneToCompare)) {
327  auto* actx = aOutsideFinality ? nullptr : a.getKeystone(keystoneToCompare);
328  auto* bctx = bOutsideFinality ? nullptr : b.getKeystone(keystoneToCompare);
329 
330  int earliestPublicationA =
331  actx == nullptr ? NO_ENDORSEMENT : actx->firstBlockPublicationHeight;
332  int earliestPublicationB =
333  bctx == nullptr ? NO_ENDORSEMENT : bctx->firstBlockPublicationHeight;
334 
335  // if keystoneToCompare is right after a gap, publicationViolatesFinality
336  // will return false, thus we only compare the finality delay violations
337  // between endorsements that exist
338  if (actx != nullptr &&
339  publicationViolatesFinality(
340  earliestPublicationA, previousPublicationA, config)) {
341  aOutsideFinality = true;
342  actx = nullptr;
343  VBK_LOG_DEBUG(
344  "Chain A is outside finality: the gap between adjacent keystone "
345  "endorsements is too large");
346  }
347  previousPublicationA = earliestPublicationA;
348 
349  if (bctx != nullptr &&
350  publicationViolatesFinality(
351  earliestPublicationB, previousPublicationB, config)) {
352  bOutsideFinality = true;
353  bctx = nullptr;
354  VBK_LOG_DEBUG(
355  "Chain B is outside finality: the gap between adjacent keystone "
356  "endorsements is too large");
357  }
358  previousPublicationB = earliestPublicationB;
359 
360  // if both chains are missing keystone at this height, ignore
361  // if both are outside finality, break
362  if (actx == nullptr && bctx == nullptr) {
363  if (aOutsideFinality && bOutsideFinality) {
364  break;
365  }
366  continue;
367  }
368 
369  if (actx == nullptr) {
370  chainBscore += config.getForkResolutionLookUpTable()[0];
371  // Nothing added to chainA; it doesn't have an endorsed keystone at
372  // this height (or any additional height)
373 
374  // chain A is missing the keystone that chain B isn't
375  aOutsideFinality = true;
376 
377  if (chainBscore > chainAscore) {
378  // exit early as the score of A will stay the same
379  break;
380  }
381 
382  continue;
383  }
384 
385  if (bctx == nullptr) {
386  chainAscore += config.getForkResolutionLookUpTable()[0];
387 
388  // chain B is missing the keystone that chain A isn't
389  bOutsideFinality = true;
390 
391  if (chainAscore > chainBscore) {
392  // exit early as the score of chain B will stay the same
393  break;
394  }
395 
396  continue;
397  }
398 
399  // both chains have a keystone at this height
400 
401  int earliestPublicationOfEither =
402  (std::min)(earliestPublicationA, earliestPublicationB);
403 
404  chainAscore += getConsensusScoreFromRelativeBlockStartingAtZero(
405  earliestPublicationA - earliestPublicationOfEither, config);
406  chainBscore += getConsensusScoreFromRelativeBlockStartingAtZero(
407  earliestPublicationB - earliestPublicationOfEither, config);
408 
409  if (publicationViolatesFinality(
410  earliestPublicationA, earliestPublicationB, config)) {
411  VBK_LOG_DEBUG("Chain A is outside finality: way behind chain B");
412  aOutsideFinality = true;
413  }
414 
415  if (publicationViolatesFinality(
416  earliestPublicationB, earliestPublicationA, config)) {
417  VBK_LOG_DEBUG("Chain B is outside finality: way behind chain A");
418  bOutsideFinality = true;
419  }
420  }
421 
422  VBK_LOG_DEBUG("ChainA score=%d, ChainB score=%d", chainAscore, chainBscore);
423  return chainAscore - chainBscore;
424 }

488  bool setState(protected_index_t& to, ValidationState& state) {
489  VBK_TRACE_ZONE_SCOPED;
490 
491  auto* currentActive = ed_.getBestChain().tip();
492  VBK_ASSERT(currentActive != nullptr && "should be bootstrapped");
493 
494  VBK_ASSERT_MSG(
495  currentActive->getHeight() + 1 ==
496  ed_.getRoot().getHeight() +
497  (typename protected_block_t::height_t)ed_.appliedBlockCount,
498  "the tree must have the best chain applied");
499 
500  if (currentActive == &to) {
501  // already at this state
502  return true;
503  }
504 
505  auto guard = ing_->deferForkResolutionGuard();
506  auto originalTip = ing_->getBestChain().tip();
507 
508  if (sm_.setState(*currentActive, to, state)) {
509  return true;
510  }
511 
512  guard.overrideDeferredForkResolution(originalTip);
513  return false;
514  }

523  std::pair<int, PopFrOutcome> comparePopScore(protected_index_t& candidate,
524  ValidationState& state) {
525  VBK_TRACE_ZONE_SCOPED;
526 
527  if (!candidate.isValid()) {
528  // if the new block is known to be invalid, we always return "A is better"
529  VBK_LOG_DEBUG("Candidate %s is invalid, the current chain wins",
530  candidate.toShortPrettyString());
531  return {1, PopFrOutcome::CANDIDATE_INVALID_CHAIN};
532  }
533  auto currentBest = ed_.getBestChain();
534  auto bestTip = currentBest.tip();
535  VBK_ASSERT(bestTip && "must be bootstrapped");
536 
537  if (bestTip == &candidate) {
538  // we are comparing the best chain to itself
539  return {1, PopFrOutcome::CANDIDATE_IS_TIP};
540  }
541 
542  if (bestTip->finalized && candidate.getHeight() <= bestTip->getHeight()) {
543  // finalized blocks can not be reorganized
544  return {1, PopFrOutcome::TIP_IS_FINAL};
545  }
546 
547  if (currentBest.contains(&candidate)) {
548  VBK_LOG_DEBUG(
549  "Candidate %s is part of the active chain, the current chain wins",
550  candidate.toShortPrettyString());
551  return {1, PopFrOutcome::CANDIDATE_PART_OF_ACTIVE_CHAIN};
552  }
553 
554  auto originalProtectingTip = ing_->getBestChain().tip();
555 
556  // candidate is on top of our best tip
557  if (candidate.getAncestor(bestTip->getHeight()) == bestTip) {
558  VBK_LOG_DEBUG("Candidate is %d blocks ahead",
559  candidate.getHeight() - bestTip->getHeight());
560 
561  auto guard = ing_->deferForkResolutionGuard();
562 
563  if (!sm_.apply(*bestTip, candidate, state)) {
564  // new chain is invalid. our current chain is definitely better.
565  VBK_LOG_DEBUG("Candidate contains INVALID command(s): %s",
566  state.toString());
567  guard.overrideDeferredForkResolution(originalProtectingTip);
568  return {1, PopFrOutcome::CANDIDATE_INVALID_PAYLOADS};
569  }
570 
571  VBK_LOG_DEBUG("Candidate contains VALID commands, chain B wins");
572  return {-1, PopFrOutcome::CANDIDATE_IS_TIP_SUCCESSOR};
573  }
574 
575  VBK_LOG_DEBUG("Doing %s POP fork resolution. Best=%s, Candidate=%s",
576  protected_block_t::name(),
577  bestTip->toShortPrettyString(),
578  candidate.toShortPrettyString());
579 
580  auto ki = ed_.getParams().getKeystoneInterval();
581  const auto* fork = findFork(currentBest, &candidate);
582  VBK_ASSERT_MSG(
583  fork != nullptr,
584  "state corruption: all blocks in a blocktree must form a tree, "
585  "thus all pairs of chains must have a fork point: chainA=%s, chainB=%s",
586  bestTip->toPrettyString(),
587  candidate.toPrettyString());
588 
589  // if block next to fork is finalized, we don't do FR, since it can not be
590  // reorganized.
591  auto* nextToFork = currentBest[fork->getHeight() + 1];
592  if (nextToFork != nullptr && nextToFork->finalized) {
593  // block `fork+1` is on active chain, and ancestor of currentBest.
594  // which means that `fork+1` can not be reorganized and candidate will
595  // never win.
596  return {1, PopFrOutcome::TIP_IS_FINAL};
597  }
598 
599  // perform multi-keystone PoP fork resolution
600  bool AcrossedKeystoneBoundary =
601  isCrossedKeystoneBoundary(fork->getHeight(), bestTip->getHeight(), ki);
602  bool BcrossedKeystoneBoundary =
603  isCrossedKeystoneBoundary(fork->getHeight(), candidate.getHeight(), ki);
604  if (!AcrossedKeystoneBoundary && !BcrossedKeystoneBoundary) {
605  // chains are equal in terms of POP
606  VBK_LOG_DEBUG(
607  "Neither chain crossed a keystone boundary: chains are equal");
608  return {0, PopFrOutcome::BOTH_DONT_CROSS_KEYSTONE_BOUNDARY};
609  }
610 
611  // [vbk fork point ... current tip]
612  ChainSlice<protected_index_t> chainA(currentBest, fork->getHeight());
613  // [vbk fork point ... new block]
614  Chain<protected_index_t> chainB(fork->getHeight(), &candidate);
615 
616  // chains are not empty and chains start at the same block
617  VBK_ASSERT(chainA.first() != nullptr && chainA.first() == chainB.first());
618 
619  // we ALWAYS compare the currently applied chain (chainA) and the candidate
620  // (chainB)
621  VBK_ASSERT(chainA.tip() == bestTip);
622 
623  // we are at chainA.
624  // apply all payloads from chain B (both chains have same first block - the
625  // fork point, so exclude it during 'apply')
626  {
627  auto guard = ing_->deferForkResolutionGuard();
628 
629  if (!sm_.apply(chainB, state)) {
630  // chain B has been unapplied and invalidated already
631  VBK_LOG_DEBUG("Chain B contains INVALID payloads, Chain A wins (%s)",
632  state.toString());
633  guard.overrideDeferredForkResolution(originalProtectingTip);
634  return {1, PopFrOutcome::CANDIDATE_INVALID_PAYLOADS};
635  }
636  }
637 
638  // now the tree contains payloads from both chains
639 
640  auto reducedPublicationViewA =
641  reduced_publication_view_t(chainA, protectedParams_, *ing_);
642  auto reducedPublicationViewB =
643  reduced_publication_view_t(chainB, protectedParams_, *ing_);
644 
645  int result = internal::comparePopScoreImpl(reducedPublicationViewA,
646  reducedPublicationViewB);
647  if (result >= 0) {
648  // chain A remains the best one. unapply B and leave A applied
649  auto guard = ing_->deferForkResolutionGuard();
650  sm_.unapply(chainB);
651  guard.overrideDeferredForkResolution(originalProtectingTip);
652  VBK_LOG_DEBUG("Chain A remains the best chain");
653  } else {
654  // chain B is better. unapply A and leave B applied
655  auto guard = ing_->deferForkResolutionGuard();
656 
657  // if part of chainB has uncertain validity(never been applied before with
658  // setState()), we have to unapply this part before unapplying chainA and
659  // try to apply the unvalidated part of chainB without any payloads from
