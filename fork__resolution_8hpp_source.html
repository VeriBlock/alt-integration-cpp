28template <typename ProtectingBlockT>
29struct ProtoKeystoneContext {
30 int blockHeight;
31 uint32_t timestampOfEndorsedBlock;
32
33 ProtoKeystoneContext(int height, int time)
34 : blockHeight(height), timestampOfEndorsedBlock(time) {}
35
36 // A map of the Endorsement blocks which reference (endorse a block header
37 // which is the represented block or references the represented block) to the
38 // index of the earliest
39 std::set<const BlockIndex<ProtectingBlockT>*> referencedByBlocks;
40};

52template <typename ConfigType>
53bool publicationViolatesFinality(int pubToCheck,
54 int base,
55 const ConfigType& config) {
56 int64_t diff = pubToCheck - base;
57 return diff > config.getFinalityDelay();
58}

60template <typename ConfigType>
61int getConsensusScoreFromRelativeBlockStartingAtZero(int64_t relativeBlock,
62 const ConfigType& config) {
63 if (relativeBlock < 0 ||
64 (size_t)relativeBlock >= config.getForkResolutionLookUpTable().size()) {
65 return 0;
66 }
67
68 return config.getForkResolutionLookUpTable()[relativeBlock];
69}

71constexpr int NO_ENDORSEMENT = (std::numeric_limits<int32_t>::max)();

73template <typename ProtectingBlockT, typename ProtectingChainParams>
74KeystoneContext getKeystoneContext(
75 const ProtoKeystoneContext<ProtectingBlockT>& pkc,
76 const BlockTree<ProtectingBlockT, ProtectingChainParams>& tree) {
77 VBK_TRACE_ZONE_SCOPED;
78
79 int earliestEndorsementIndex = NO_ENDORSEMENT;
80 for (const auto* btcIndex : pkc.referencedByBlocks) {
81 if (btcIndex == nullptr) {
82 continue;
83 }
84
85 auto endorsementIndex = btcIndex->getHeight();
86 if (endorsementIndex >= earliestEndorsementIndex) {
87 continue;
88 }
89
90 bool EnableTimeAdjustment = tree.getParams().EnableTimeAdjustment();
91 if (!EnableTimeAdjustment ||
92 pkc.timestampOfEndorsedBlock < btcIndex->getTimestamp()) {
93 earliestEndorsementIndex = endorsementIndex;
94 continue;
95 }
96
97 // look at the future BTC blocks and set the
98 // earliestEndorsementIndex to a future Bitcoin block
99 auto best = tree.getBestChain();
100 for (int adjustedEndorsementIndex = endorsementIndex + 1;
101 adjustedEndorsementIndex <= best.chainHeight();
102 adjustedEndorsementIndex++) {
103 // Ensure that the keystone's block time isn't later than the
104 // block time of the Bitcoin block it's endorsed in
105 auto* index = best[adjustedEndorsementIndex];
106 VBK_ASSERT(index != nullptr);
107 if (pkc.timestampOfEndorsedBlock < index->getTimestamp()) {
108 // Timestamp of VeriBlock block is lower than Bitcoin block,
109 // set this as the adjusted index if another lower index has
110 // not already been set
111 if (adjustedEndorsementIndex < earliestEndorsementIndex) {
112 earliestEndorsementIndex = adjustedEndorsementIndex;
113 }
114
115 // Always break; we found a valid Bitcoin index and any
116 // future adjustedEndorsementIndex is going to be higher
117 break;
118 } // end if
119 } // end for
120 } // end for
121
122 return {pkc.blockHeight, earliestEndorsementIndex};
123}

125template <typename ProtectedBlockT,
126 typename ProtectingBlockT,
127 typename ProtectingChainParams,
128 typename ProtectedChainParams,
129 typename ProtectedBlockTree>
130ProtoKeystoneContext<ProtectingBlockT> getProtoKeystoneContext(
131 int keystoneToConsider,
132 const ChainSlice<BlockIndex<ProtectedBlockT>>& chain,
133 const ProtectedBlockTree& ed,
134 const BlockTree<ProtectingBlockT, ProtectingChainParams>& ing,
135 const ProtectedChainParams& config) {
136 VBK_TRACE_ZONE_SCOPED;
137 auto ki = config.getKeystoneInterval();
138 auto* tip = chain.tip();
139 VBK_ASSERT(tip != nullptr && "chain must not be empty");
140
141 auto highestConnectingBlock =
142 highestBlockWhichConnectsKeystoneToPrevious(keystoneToConsider, ki);
143 auto highestPossibleEndorsedBlockHeaderHeight = tip->getHeight();
144 auto highestEndorsedBlock = std::min(
145 highestConnectingBlock, highestPossibleEndorsedBlockHeaderHeight);
146
147 ProtoKeystoneContext<ProtectingBlockT> pkc(
148 keystoneToConsider, chain[keystoneToConsider]->getTimestamp());
149
150 // Find the endorsements of the keystone block and other blocks which
151 // reference it, and look at the earliest Bitcoin block that any of those
152 // endorsements are contained within.
153 for (auto relevantEndorsedBlock = keystoneToConsider;
154 relevantEndorsedBlock <= highestEndorsedBlock;
155 relevantEndorsedBlock++) {
156 auto* index = chain[relevantEndorsedBlock];
157
158 // chain must contain relevantEndorsedBlock
159 VBK_ASSERT(index != nullptr);
160
161 for (const auto* e : index->getEndorsedBy()) {
162 VBK_ASSERT(e != nullptr);
163 auto* containingBlock = ed.getBlockIndex(e->containingHash);
164 VBK_ASSERT_MSG(containingBlock != nullptr,
165 "state corruption: could not find the containing block of "
166 "an applied endorsement");
167
168 if (!chain.contains(containingBlock)) {
169 // do not count endorsement whose containingHash is not on the same
170 // chain as 'endorsedHash'
171 continue;
172 }
173
174 auto* ind = ing.getBlockIndex(e->blockOfProof);
175 VBK_ASSERT(ind != nullptr &&
176 "state corruption: could not find the block of proof of "
177 "an applied endorsement");
178 if (!ing.getBestChain().contains(ind)) {
179 continue;
180 }
181
182 // include only endorsements that are on best chain of protecting chain,
183 // and whose 'containingHash' is on the same chain as 'endorsedHash'
184 pkc.referencedByBlocks.insert(ind);
185
186 } // end for
187 } // end for
188
189 return pkc;
190}

194template <typename ProtectedBlockT,
195 typename ProtectingBlockT,
196 typename ProtectingChainParams,
197 typename ProtectedChainParams,

199struct ReducedPublicationView {
200 using protecting_block_t = ProtectingBlockT;
201 using protected_block_t = ProtectedBlockT;
202 using protecting_chain_params_t = ProtectingChainParams;
203 using protected_chain_params_t = ProtectedChainParams;
204 using protected_chain_t = ChainSlice<BlockIndex<protected_block_t>>;
205 using protected_tree_t = ProtectedBlockTree;
206 using protecting_tree_t =
207 BlockTree<protecting_block_t, protecting_chain_params_t>;
208
209 const protected_chain_params_t& config;
210 const int keystoneInterval;
211 const protected_chain_t chain;
212
213 const protected_tree_t& protectedTree;
214 const protecting_tree_t& protectingTree;
215
216 const int firstKeystoneHeight;
217 const int lastKeystoneHeight;
218
219 KeystoneContext currentKeystoneContext; // FIXME: UGLY
220
221 ReducedPublicationView(const protected_chain_t _chain,
222 const protected_chain_params_t& _config,
223 const protected_tree_t& _ed,
224 const protecting_tree_t& _ing)
225 : config(_config),
226 keystoneInterval(config.getKeystoneInterval()),
227 chain(_chain),
228 protectedTree(_ed),
229 protectingTree(_ing),
230 firstKeystoneHeight(firstKeystoneAfterBlockIndex(chain.first())),
231 lastKeystoneHeight(highestKeystoneAtOrBeforeBlockIndex(chain.tip())) {
232 VBK_ASSERT(keystoneInterval > 0);
233 }
234
235 const protected_chain_params_t& getConfig() const { return config; }
236
237 size_t size() const {
238 return blockHeightToKeystoneNumber(lastKeystone(), keystoneInterval) -
239 blockHeightToKeystoneNumber(firstKeystone(), keystoneInterval) + 1;
240 }
241
242 bool empty() const { return size() == 0; }
243
244 int firstKeystone() const { return firstKeystoneHeight; }
245
246 int lastKeystone() const { return lastKeystoneHeight; }
247
248 int nextKeystoneAfter(int keystoneHeight) const {
249 VBK_ASSERT(isKeystone(keystoneHeight, keystoneInterval));
250 return keystoneHeight + keystoneInterval;
251 }
252
253 const KeystoneContext* operator[](int blockHeight) {
254 return getKeystone(blockHeight);
255 }
256
257 const KeystoneContext* getKeystone(int blockHeight) {
258 VBK_ASSERT_MSG(
259 isKeystone(blockHeight, keystoneInterval),
260 "getKeystone can not be called with a non-keystone block height");
261
262 if (blockHeight < firstKeystone() || blockHeight > lastKeystone()) {
263 return nullptr;
264 }
265
266 // FIXME: there's no need to store the intermediate list of blocks and thus
267 // no need for ProtoKeystoneContext entity
268 auto pkc = getProtoKeystoneContext(
269 blockHeight, chain, protectedTree, protectingTree, config);
270
271 currentKeystoneContext = getKeystoneContext(pkc, protectingTree);
272 return &currentKeystoneContext;
273 }
274
275 int32_t firstKeystoneAfterBlockIndex(
276 typename protected_chain_t::index_t* blockIndex) {
277 VBK_ASSERT_MSG(blockIndex, "Protected tree should be bootstrapped");
278 return firstKeystoneAfter(blockIndex->getHeight(), keystoneInterval);
279 }
280
281 int32_t highestKeystoneAtOrBeforeBlockIndex(
282 typename protected_chain_t::index_t* blockIndex) {
283 VBK_ASSERT_MSG(blockIndex, "Protected tree should be bootstrapped");
284 return highestKeystoneAtOrBefore(blockIndex->getHeight(), keystoneInterval);
285 }

288template <typename PublicationView>
289int comparePopScoreImpl(PublicationView& a, PublicationView& b) {
290 VBK_TRACE_ZONE_SCOPED;
291
292 if (a.empty() && b.empty()) {
293 return 0;
294 }
295
296 if (a.empty()) {
297 // a is empty, b is not
298 return -1;
299 }
300
301 if (b.empty()) {
302 // b is empty, a is not
303 return 1;
304 }
305
306 VBK_LOG_DEBUG(
307 "Comparing POP scores of chains A(first=%d, tip=%d) "
308 "and B(first=%d, tip=%d)",
309 a.firstKeystone(),
310 a.lastKeystone(),
311 b.firstKeystone(),
312 b.lastKeystone());
313
314 int earliestKeystone = a.firstKeystone();
315 VBK_ASSERT(earliestKeystone == b.firstKeystone());
316 int latestKeystone = (std::max)(a.lastKeystone(), b.lastKeystone());
317
318 auto& config = a.getConfig();
319 VBK_ASSERT(&config == &b.getConfig()); // FIXME: comparing pointers is UGLY
320 // clang-format off
321 // If either chain has a keystone the other chain is missing, the other chain
322 // goes outside finality.
323 //
324 // Example:
325 // Chain A has keystones VBK20:BTC100, VBK40:BTC101, VBK60:BTC103, VBK100:BTC104
326 // Chain B has keystones VBK20:BTC100, VBK40:BTC101, VBK100:BTC102
327 //
328 // Chain A has keystone 60 but chain B does not, so Chain B is
329 // chopped to VBK20:BTC100, VBK40:BTC101.
330
331 // Also, the chain goes out of finality if it contains a keystone which violates
332 // the Bitcoin finality delay based on the previous keystone in the chain.
333 // clang-format on
334
335 bool aOutsideFinality = false;
336 bool bOutsideFinality = false;
337 int chainAscore = 0;
338 int chainBscore = 0;
339 int previousPublicationA = NO_ENDORSEMENT;
340 int previousPublicationB = NO_ENDORSEMENT;
341
342 for (int keystoneToCompare = earliestKeystone;
343 keystoneToCompare <= latestKeystone;
344 keystoneToCompare = a.nextKeystoneAfter(keystoneToCompare)) {
345 auto* actx = aOutsideFinality ? nullptr : a.getKeystone(keystoneToCompare);
346 auto* bctx = bOutsideFinality ? nullptr : b.getKeystone(keystoneToCompare);
347
348 int earliestPublicationA =
349 actx == nullptr ? NO_ENDORSEMENT : actx->firstBlockPublicationHeight;
350 int earliestPublicationB =
351 bctx == nullptr ? NO_ENDORSEMENT : bctx->firstBlockPublicationHeight;
352
353 // if keystoneToCompare is right after a gap, publicationViolatesFinality
354 // will return false, thus we only compare the finality delay violations
355 // between endorsements that exist
356 if (actx != nullptr &&
357 publicationViolatesFinality(
358 earliestPublicationA, previousPublicationA, config)) {
359 aOutsideFinality = true;
360 actx = nullptr;
361 VBK_LOG_DEBUG(
362 "Chain A is outside finality: the gap between adjacent keystone "
363 "endorsements is too large");
364 }
365 previousPublicationA = earliestPublicationA;
366
367 if (bctx != nullptr &&
368 publicationViolatesFinality(
369 earliestPublicationB, previousPublicationB, config)) {
370 bOutsideFinality = true;
371 bctx = nullptr;
372 VBK_LOG_DEBUG(
373 "Chain B is outside finality: the gap between adjacent keystone "
374 "endorsements is too large");
375 }
376 previousPublicationB = earliestPublicationB;
377
378 // if both chains are missing keystone at this height, ignore
379 // if both are outside finality, break
380 if (actx == nullptr && bctx == nullptr) {
381 if (aOutsideFinality && bOutsideFinality) {
382 break;
383 }
384 continue;
385 }
386
387 if (actx == nullptr) {
388 chainBscore += config.getForkResolutionLookUpTable()[0];
389 // Nothing added to chainA; it doesn't have an endorsed keystone at
390 // this height (or any additional height)
391
392 // chain A is missing the keystone that chain B isn't
393 aOutsideFinality = true;
394
395 if (chainBscore > chainAscore) {
396 // exit early as the score of A will stay the same
397 break;
398 }
399
400 continue;
401 }
402
403 if (bctx == nullptr) {
404 chainAscore += config.getForkResolutionLookUpTable()[0];
405
406 // chain B is missing the keystone that chain A isn't
407 bOutsideFinality = true;
408
409 if (chainAscore > chainBscore) {
410 // exit early as the score of chain B will stay the same
411 break;
412 }
413
414 continue;
415 }
416
417 // both chains have a keystone at this height
418
419 int earliestPublicationOfEither =
420 (std::min)(earliestPublicationA, earliestPublicationB);
421
422 chainAscore += getConsensusScoreFromRelativeBlockStartingAtZero(
423 earliestPublicationA - earliestPublicationOfEither, config);
424 chainBscore += getConsensusScoreFromRelativeBlockStartingAtZero(
425 earliestPublicationB - earliestPublicationOfEither, config);
426
427 if (publicationViolatesFinality(
428 earliestPublicationA, earliestPublicationB, config)) {
429 VBK_LOG_DEBUG("Chain A is outside finality: way behind chain B");
430 aOutsideFinality = true;
431 }
432
433 if (publicationViolatesFinality(
434 earliestPublicationB, earliestPublicationA, config)) {
435 VBK_LOG_DEBUG("Chain B is outside finality: way behind chain A");
436 bOutsideFinality = true;
437 }
438 }
439
440 VBK_LOG_DEBUG("ChainA score=%d, ChainB score=%d", chainAscore, chainBscore);
441 return chainAscore - chainBscore;
442}

488 PopAwareForkResolutionComparator(ProtectedBlockTree& ed,
489 std::shared_ptr<ProtectingBlockTree> ing,
490 const protected_params_t& protectedParams,
491 PayloadsStorage& payloadsProvider,
492 PayloadsIndex& payloadsIndex)
493 : ed_(ed),
494 ing_(std::move(ing)),
495 protectedParams_(protectedParams),
496 payloadsProvider_(payloadsProvider),
497 payloadsIndex_(payloadsIndex),
498 sm_(ed_, *ing_, payloadsIndex_) {
499 VBK_ASSERT(protectedParams.getKeystoneInterval() > 0);
500 }

502 ProtectingBlockTree& getProtectingBlockTree() { return *ing_; }
503 const ProtectingBlockTree& getProtectingBlockTree() const { return *ing_; }

509 bool setState(protected_index_t& to, ValidationState& state) {
510 VBK_TRACE_ZONE_SCOPED;
511
512 auto* currentActive = ed_.getBestChain().tip();
513 VBK_ASSERT(currentActive != nullptr && "should be bootstrapped");
514
515 VBK_ASSERT_MSG(
516 currentActive->getHeight() + 1 ==
517 ed_.getRoot().getHeight() +
518 (typename protected_block_t::height_t)ed_.appliedBlockCount,
519 "the tree must have the best chain applied");
520
521 if (currentActive == &to) {
522 // already at this state
523 return true;
524 }
525
526 auto guard = ing_->deferForkResolutionGuard();
527 auto originalTip = ing_->getBestChain().tip();
528
529 if (sm_.setState(*currentActive, to, state)) {
530 return true;
531 }
532
533 guard.overrideDeferredForkResolution(originalTip);
534 return false;
535 }

544 std::pair<int, PopFrOutcome> comparePopScore(protected_index_t& candidate,
545 ValidationState& state) {
546 VBK_TRACE_ZONE_SCOPED;
547
548 if (!candidate.isValid()) {
549 // if the new block is known to be invalid, we always return "A is better"
550 VBK_LOG_DEBUG("Candidate %s is invalid, the current chain wins",
551 candidate.toShortPrettyString());
552 return {1, PopFrOutcome::CANDIDATE_INVALID_CHAIN};
553 }
554 auto currentBest = ed_.getBestChain();
555 auto bestTip = currentBest.tip();
556 VBK_ASSERT(bestTip && "must be bootstrapped");
557
558 if (bestTip == &candidate) {
559 // we are comparing the best chain to itself
560 return {1, PopFrOutcome::CANDIDATE_IS_TIP};
561 }
562
563 if (bestTip->finalized && candidate.getHeight() <= bestTip->getHeight()) {
564 // finalized blocks can not be reorganized
565 return {1, PopFrOutcome::TIP_IS_FINAL};
566 }
567
568 if (currentBest.contains(&candidate)) {
569 VBK_LOG_DEBUG(
570 "Candidate %s is part of the active chain, the current chain wins",
571 candidate.toShortPrettyString());
572 return {1, PopFrOutcome::CANDIDATE_PART_OF_ACTIVE_CHAIN};
573 }
574
575 auto originalProtectingTip = ing_->getBestChain().tip();
576
577 // candidate is on top of our best tip
578 if (candidate.getAncestor(bestTip->getHeight()) == bestTip) {
579 VBK_LOG_DEBUG("Candidate is %d blocks ahead",
580 candidate.getHeight() - bestTip->getHeight());
581
582 auto guard = ing_->deferForkResolutionGuard();
583
584 if (!sm_.apply(*bestTip, candidate, state)) {
585 // new chain is invalid. our current chain is definitely better.
586 VBK_LOG_DEBUG("Candidate contains INVALID command(s): %s",
587 state.toString());
588 guard.overrideDeferredForkResolution(originalProtectingTip);
589 return {1, PopFrOutcome::CANDIDATE_INVALID_PAYLOADS};
590 }
591
592 VBK_LOG_DEBUG("Candidate contains VALID commands, chain B wins");
593 return {-1, PopFrOutcome::CANDIDATE_IS_TIP_SUCCESSOR};
594 }
595
596 VBK_LOG_DEBUG("Doing %s POP fork resolution. Best=%s, Candidate=%s",
597 protected_block_t::name(),
598 bestTip->toShortPrettyString(),
599 candidate.toShortPrettyString());
600
601 auto ki = ed_.getParams().getKeystoneInterval();
602 const auto* fork = findFork(currentBest, &candidate);
603 VBK_ASSERT_MSG(
604 fork != nullptr,
605 "state corruption: all blocks in a blocktree must form a tree, "
606 "thus all pairs of chains must have a fork point: chainA=%s, chainB=%s",
607 bestTip->toPrettyString(),
608 candidate.toPrettyString());
609
610 // if block next to fork (on active chain) is finalized, we don't do FR,
611 // since it can not be reorganized.
612 auto* nextToFork = currentBest[fork->getHeight() + 1];
613 if (nextToFork != nullptr && nextToFork->finalized) {
614 // block `fork+1` is on active chain, and ancestor of currentBest.
615 // which means that `fork+1` can not be reorganized and candidate will
616 // never win.
617 return {1, PopFrOutcome::TIP_IS_FINAL};
618 }
619
620 // perform multi-keystone PoP fork resolution
621 bool AcrossedKeystoneBoundary =
622 isCrossedKeystoneBoundary(fork->getHeight(), bestTip->getHeight(), ki);
623 bool BcrossedKeystoneBoundary =
624 isCrossedKeystoneBoundary(fork->getHeight(), candidate.getHeight(), ki);
625 if (!AcrossedKeystoneBoundary && !BcrossedKeystoneBoundary) {
626 // chains are equal in terms of POP
627 VBK_LOG_DEBUG(
628 "Neither chain crossed a keystone boundary: chains are equal");
629 return {0, PopFrOutcome::BOTH_DONT_CROSS_KEYSTONE_BOUNDARY};
630 }
631
632 // [vbk fork point ... current tip]
633 ChainSlice<protected_index_t> chainA(currentBest, fork->getHeight());
634 // [vbk fork point ... new block]
635 Chain<protected_index_t> chainB(fork->getHeight(), &candidate);
636
637 // chains are not empty and chains start at the same block
638 VBK_ASSERT(chainA.first() != nullptr && chainA.first() == chainB.first());
639
640 // we ALWAYS compare the currently applied chain (chainA) and the candidate
641 // (chainB)
642 VBK_ASSERT(chainA.tip() == bestTip);
643
644 // we are at chainA.
645 // apply all payloads from chain B (both chains have same first block - the
646 // fork point, so exclude it during 'apply')
647 {
648 auto guard = ing_->deferForkResolutionGuard();
649
650 if (!sm_.apply(chainB, state)) {
651 // chain B has been unapplied and invalidated already
652 VBK_LOG_DEBUG("Chain B contains INVALID payloads, Chain A wins (%s)",
653 state.toString());
654 guard.overrideDeferredForkResolution(originalProtectingTip);
655 return {1, PopFrOutcome::CANDIDATE_INVALID_PAYLOADS};
656 }
657 }
658
659 // now the tree contains payloads from both chains
660 auto reducedPublicationViewA =
661 reduced_publication_view_t(chainA, protectedParams_, ed_, *ing_);
662 auto reducedPublicationViewB =
663 reduced_publication_view_t(chainB, protectedParams_, ed_, *ing_);
664
665 int result = internal::comparePopScoreImpl(reducedPublicationViewA,
666 reducedPublicationViewB);
667 if (result >= 0) {
668 // chain A remains the best one. unapply B and leave A applied
669 auto guard = ing_->deferForkResolutionGuard();
670 sm_.unapply(chainB);
671 guard.overrideDeferredForkResolution(originalProtectingTip);
672 VBK_LOG_DEBUG("Chain A remains the best chain");
673 } else {
674 // chain B is better. unapply A and leave B applied
675 auto guard = ing_->deferForkResolutionGuard();
676
677 // if part of chainB has uncertain validity(never been applied before with
678 // setState()), we have to unapply this part before unapplying chainA and
679 // try to apply the unvalidated part of chainB without any payloads from
680 // chainA to make sure it is fully valid
681 auto& chainBValidFrom =
682 sm_.unapplyWhile(chainB, [](protected_index_t& index) -> bool {
683 return !index.isValid(BLOCK_CAN_BE_APPLIED);
684 });
685
686 // unapply losing chain
687 sm_.unapply(chainA);
688
689 // validate the unvalidated part of chainB
690 if (!sm_.apply(chainBValidFrom, *chainB.tip(), state)) {
691 sm_.unapply(chainBValidFrom, *chainB.first());
692 bool success = sm_.apply(*chainA.first(), *chainA.tip(), state);
693 VBK_ASSERT_MSG(
694 success,
695 "state corruption: chainA as a former best chain should be valid");
696 VBK_LOG_DEBUG("Chain B is invalid when applied alone. Chain A wins");
697 guard.overrideDeferredForkResolution(originalProtectingTip);
698 return {1, PopFrOutcome::CANDIDATE_INVALID_INDEPENDENTLY};
699 }
