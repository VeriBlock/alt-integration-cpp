11 #ifdef VBK_HAS_BUILTIN_EXPECT
12 // tell branch predictor that condition is always true
13 #define VBK_LIKELY(condition) __builtin_expect(static_cast<bool>(condition), 1)
14 #define VBK_UNLIKELY(condition) \
15  __builtin_expect(static_cast<bool>(condition), 0)
16 #else
17 #define VBK_LIKELY(condition) (condition)
18 #define VBK_UNLIKELY(condition) (condition)
19 #endif

37 #define VBK_ASSERT_MSG(x, ...) \
38  /* GCOVR_EXCL_START */ \
39  if (!VBK_LIKELY((x))) { \
40  auto msg = fmt::format("Assertion failed at {}:{} inside {}:\n{}\n{}\n", \
41  __FILE__, \
42  __LINE__, \
43  __FUNCTION__, \
44  #x, \
45  fmt::sprintf(__VA_ARGS__)); \
46  /* print to log */ VBK_LOG_CRITICAL(msg); \
47  /* print to stderr */ fmt::fprintf(stderr, msg); \
48  /* die */ std::terminate(); \
49  } /* GCOVR_EXCL_STOP */
50 
51 #define VBK_ASSERT(x) VBK_ASSERT_MSG(x, " ");
52 
53 // same as VBK_ASSERT but executes only when compiled in DEBUG mode. similar to

58 #else
59 #define VBK_ASSERT_MSG_DEBUG(x, ...) VBK_ASSERT_MSG(x, __VA_ARGS__)
60 #define VBK_ASSERT_DEBUG(x) VBK_ASSERT(x)
61 #endif
