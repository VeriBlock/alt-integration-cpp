96template <typename Container,
97 typename = typename std::enable_if<
98 sizeof(typename Container::value_type) == 1>::type>
99bool readSingleByteLenValue(ReadStream& stream,
100 Container& out,
101 ValidationState& state,
102 size_t minLen,
103 size_t maxLen) {
104 uint8_t length = 0;
105 if (!stream.readBE<uint8_t>(length, state)) {
106 return state.Invalid("readsingle-bad-length");
107 }
108 if (!checkRange(length, minLen, maxLen, state)) {
109 return state.Invalid("readsingle-bad-range");
110 }
111 out.resize(length);
112 return stream.read(length, out.data(), state);
113}

125template <typename T,
126 typename = typename std::enable_if<std::is_integral<T>::value>::type>
127bool readSingleBEValue(ReadStream& stream, T& out, ValidationState& state) {
128 Slice<const uint8_t> data;
129 if (!readSingleByteLenValue(stream, data, state, 0, sizeof(T))) {
130 return state.Invalid("readsinglebe-bad-data");
131 }
132 ReadStream dataStream(data);
133 return dataStream.readBE<T>(out, state, data.size());
134}

252template <typename T>
253bool readArrayOf(
254 ReadStream& stream,
255 std::vector<T>& out,
256 ValidationState& state,
257 size_t min,
258 size_t max,
259 std::function<bool(ReadStream&, T&, ValidationState&)> readFunc) {
260 int32_t count = 0;
261 if (!readSingleBEValue<int32_t>(stream, count, state)) {
262 return state.Invalid("readarray-bad-count");
263 }
264 if (!checkRange(count, min, max, state)) {
265 return state.Invalid("readarray-bad-range");
266 }
267
268 out.reserve(count);
269
270 for (size_t i = 0; i < (size_t)count; i++) {
271 T item;
272 if (!readFunc(stream, item, state)) {
273 return state.Invalid("readarray-bad-item", i);
274 }
275 out.push_back(item);
276 }
277
278 return true;
279}

302template <typename Container>
303void writeContainer(
304 WriteStream& w,
305 const Container& t,
306 std::function<void(WriteStream&, const typename Container::value_type& t)>
307 f) {
308 writeSingleBEValue(w, (int64_t)t.size());
309 for (auto& v : t) {
310 f(w, v);
311 }
312}

323template <typename Container>
324size_t estimateContainerSize(
325 const Container& t,
326 std::function<size_t(const typename Container::value_type& t)> f) {
327 size_t size = 0;
328 size += singleBEValueSize((int64_t)t.size());
329 for (auto& v : t) {
