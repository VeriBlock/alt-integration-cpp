106template <typename Container,
107 typename = typename std::enable_if<
108 sizeof(typename Container::value_type) == 1>::type>
109bool readSingleByteLenValue(ReadStream& stream,
110 Container& out,
111 ValidationState& state,
112 uint64_t minLen,
113 uint64_t maxLen) {
114 uint8_t length = 0;
115 if (!stream.readBE<uint8_t>(length, state)) {
116 return state.Invalid("readsingle-bad-length");
117 }
118 if (!checkRange(length, minLen, maxLen, state)) {
119 return state.Invalid("readsingle-bad-range");
120 }
121 out.resize(length);
122 return stream.read(length, out.data(), state);
123}

135template <typename T,
136 typename = typename std::enable_if<std::is_integral<T>::value>::type>
137bool readSingleBEValue(ReadStream& stream, T& out, ValidationState& state) {
138 Slice<const uint8_t> data;
139 if (!readSingleByteLenValue(stream, data, state, 0, sizeof(T))) {
140 return state.Invalid("readsinglebe-bad-data");
141 }
142 ReadStream dataStream(data);
143 return dataStream.readBE<T>(out, state, data.size());
144}

262template <typename Container>
263bool readSetOf(ReadStream& stream,
264 Container& out,
265 ValidationState& state,
266 size_t min,
267 size_t max,
268 std::function<bool(ReadStream&,
269 typename Container::value_type&,
270 ValidationState&)> readFunc) {
271 int32_t count = 0;
272 if (!readSingleBEValue<int32_t>(stream, count, state)) {
273 return state.Invalid("readarray-bad-count");
274 }
275 if (!checkRange(count, min, max, state)) {
276 return state.Invalid("readarray-bad-range");
277 }
278
279 for (size_t i = 0; i < (size_t)count; i++) {
280 typename Container::value_type item;
281 if (!readFunc(stream, item, state)) {
282 return state.Invalid("readarray-bad-item", i);
283 }
284 if (out.insert(item).second == false) {
285 return state.Invalid("readarray-duplicate-item", i);
286 }
287 }
288
289 return true;
290}

325template <typename T>
326bool readArrayOf(
327 ReadStream& stream,
328 std::vector<T>& out,
329 ValidationState& state,
330 size_t min,
331 size_t max,
332 std::function<bool(ReadStream&, T&, ValidationState&)> readFunc) {
333 int32_t count = 0;
334 if (!readSingleBEValue<int32_t>(stream, count, state)) {
335 return state.Invalid("readarray-bad-count");
336 }
337 if (!checkRange(count, min, max, state)) {
338 return state.Invalid("readarray-bad-range");
339 }
340
341 out.reserve(count);
342
343 for (size_t i = 0; i < (size_t)count; i++) {
344 T item;
345 if (!readFunc(stream, item, state)) {
346 return state.Invalid("readarray-bad-item", i);
347 }
348 out.push_back(item);
349 }
350
351 return true;
352}

375template <typename Container>
376void writeContainer(
377 WriteStream& w,
378 const Container& t,
379 std::function<void(WriteStream&, const typename Container::value_type& t)>
380 f) {
381 writeSingleBEValue(w, (int64_t)t.size());
382 for (auto& v : t) {
383 f(w, v);
384 }
385}
386
388template <typename Container>
389size_t estimateContainerSize(
390 const Container& t,
391 std::function<size_t(const typename Container::value_type& t)> f) {
392 size_t size = 0;
393 size += singleBEValueSize((int64_t)t.size());
394 for (auto& v : t) {
395 size += f(v);
396 }
397 return size;
398}
