94 template <typename Container,
95  typename = typename std::enable_if<
96  sizeof(typename Container::value_type) == 1>::type>
97 bool readSingleByteLenValue(ReadStream& stream,
98  Container& out,
99  ValidationState& state,
100  size_t minLen,
101  size_t maxLen) {
102  uint8_t length = 0;
103  if (!stream.readBE<uint8_t>(length, state)) {
104  return state.Invalid("readsingle-bad-length");
105  }
106  if (!checkRange(length, minLen, maxLen, state)) {
107  return state.Invalid("readsingle-bad-range");
108  }
109  out.resize(length);
110  return stream.read(length, out.data(), state);
111 }

250 template <typename T>
251 bool readArrayOf(
252  ReadStream& stream,
253  std::vector<T>& out,
254  ValidationState& state,
255  size_t min,
256  size_t max,
257  std::function<bool(ReadStream&, T&, ValidationState&)> readFunc) {
258  int32_t count = 0;
259  if (!readSingleBEValue<int32_t>(stream, count, state)) {
260  return state.Invalid("readarray-bad-count");
261  }
262  if (!checkRange(count, min, max, state)) {
263  return state.Invalid("readarray-bad-range");
264  }
265 
266  out.reserve(count);
267 
268  for (size_t i = 0; i < (size_t)count; i++) {
269  T item;
270  if (!readFunc(stream, item, state)) {
271  return state.Invalid("readarray-bad-item", i);
272  }
273  out.push_back(item);
274  }
275 
276  return true;
277 }
