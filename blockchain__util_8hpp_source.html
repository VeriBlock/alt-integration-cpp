57 template <typename ProtectedBlockTree>
58 bool recoverEndorsements(ProtectedBlockTree& ed_,
59  Chain<typename ProtectedBlockTree::index_t>& chain,
60  typename ProtectedBlockTree::index_t& toRecover,
61  ValidationState& state) {
62  std::vector<std::function<void()>> actions;
63  auto& containingEndorsements = toRecover.getContainingEndorsements();
64  actions.reserve(containingEndorsements.size());
65  auto& ing = ed_.getComparator().getProtectingBlockTree();
66 
67  for (const auto& p : containingEndorsements) {
68  auto& id = p.first;
69  auto& e = *p.second;
70 
71  if (id != e.id) {
72  return state.Invalid("bad-id",
73  format("Key={}, Id={}", HexStr(id), HexStr(e.id)));
74  }
75 
76  auto* endorsed = ed_.getBlockIndex(e.endorsedHash);
77  if (!endorsed) {
78  return state.Invalid(
79  "no-endorsed",
80  format("Can not find endorsed block in {}", e.toPrettyString()));
81  }
82 
83  if (chain[endorsed->getHeight()] == nullptr ||
84  chain[endorsed->getHeight()]->getHash() != e.endorsedHash ||
85  endorsed->getHash() != e.endorsedHash) {
86  return state.Invalid(
87  "bad-endorsed",
88  format("Endorsed block does not match {}", e.toPrettyString()));
89  }
90 
91  if (e.containingHash != toRecover.getHash()) {
92  return state.Invalid(
93  "bad-containing",
94  format("Containing block does not match {}", e.toPrettyString()));
95  }
96 
97  auto* blockOfProof = as_mut(ing).getBlockIndex(e.blockOfProof);
98  if (!blockOfProof) {
99  return state.Invalid(
100  "bad-blockofproof",
101  format("Block Of Proof %s does not exist in SP chain",
102  HexStr(e.blockOfProof)));
103  }
104 
105  // make sure it is accessible in lambda
106  const auto* endorsement = &e;
107 
108  // delay execution. this ensures atomic changes - if any of endorsemens fail
109  // validation, no 'action' is actually executed.
110  actions.push_back([endorsed, blockOfProof, endorsement] {
111  auto& by = endorsed->getEndorsedBy();
112  VBK_ASSERT_MSG(std::find(by.begin(), by.end(), endorsement) == by.end(),
113  "same endorsement is added to endorsedBy second time");
114  endorsed->insertEndorsedBy(endorsement);
115 
116  const auto& bop = blockOfProof->getBlockOfProofEndorsement();
117  VBK_ASSERT_MSG(
118  std::find(bop.begin(), bop.end(), endorsement) == bop.end(),
119  "same endorsement is added to blockOfProof second time");
120  blockOfProof->insertBlockOfProofEndorsement(endorsement);
121  });
122  }
123 
124  // all actions have been validated
125  // commit changes
126  for (auto& f : actions) {
127  f();
128  }
129 
130  return true;
131 }

193  cgs.push_back(std::move(cg));

198 template <typename BlockTree>
199 bool areOnSameChain(const typename BlockTree::block_t& blk1,
200  const typename BlockTree::block_t& blk2,
201  const BlockTree& tree) {
202  auto* blk_index1 = tree.getBlockIndex(blk1.getHash());
203  auto* blk_index2 = tree.getBlockIndex(blk2.getHash());
204 
205  VBK_ASSERT_MSG(blk_index1, "unknown block %s", blk1.toPrettyString());
206  VBK_ASSERT_MSG(blk_index2, "unknown block %s", blk2.toPrettyString());
207 
208  if (blk_index1->getHeight() > blk_index2->getHeight()) {
209  return blk_index1->getAncestor(blk_index2->getHeight()) == blk_index2;
210  } else {
211  return blk_index2->getAncestor(blk_index1->getHeight()) == blk_index1;
212  }
213 }
