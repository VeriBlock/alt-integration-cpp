191  bool restoreBlock(const typename block_t::hash_t& hash,
192  ValidationState& state) {
193  VBK_TRACE_ZONE_SCOPED;
194 
195  index_t* root = &this->getRoot();
196  auto oldHeight = root->getHeight();
197  std::vector<stored_index_t> tempChain;
198  auto restoringHash = this->makePrevHash(hash);
199  index_t* restoringIndex = this->getBlockIndex(restoringHash);
200 
201  while (restoringIndex == nullptr) {
202  // load current block from storage
203  stored_index_t restoringBlock;
204  if (!this->blockProvider_.getBlock(restoringHash, restoringBlock)) {
205  return state.Invalid("can-not-find-block-in-storage");
206  }
207 
208  // restore previous root blocks
209  while ((restoringBlock.height < root->getHeight()) &&
210  (root->getHeight() > 0)) {
211  stored_index_t tmpRoot;
212  auto rootPrevHash = root->getHeader().getPreviousBlock();
213  if (!this->blockProvider_.getBlock(rootPrevHash, tmpRoot)) {
214  VBK_ASSERT_MSG(false,
215  "can not restore prev block for the block: %s",
216  root->toPrettyString());
217  }
218 
219  if (!loadBlockBackward(tmpRoot, state)) {
220  VBK_ASSERT_MSG(
221  false, "can not load block, state: %s", state.toString());
222  }
223 
224  root = this->getBlockIndex(rootPrevHash);
225  VBK_ASSERT(root);
226  }
227 
228  restoringIndex = getBlockIndex(restoringHash);
229  // add unconnected blocks to the temporary chain
230  if (restoringIndex == nullptr) {
231  // could not find the common block
232  if (restoringBlock.height == 0) {
233  return state.Invalid("restored-block-does-not-connect");
234  }
235 
236  tempChain.push_back(restoringBlock);
237  }
238 
239  // get the previous block hash
240  restoringHash = restoringBlock.header->getPreviousBlock();
241  }
242 
243  for (const auto& b : reverse_iterate(tempChain)) {
244  if (!loadBlockForward(b, state)) {
245  return state.Invalid("load-block-forward-failed");
246  }
247  }
248 
249  increaseAppliedBlockCount(oldHeight - this->getRoot().getHeight());
250 
251  return true;
252  }

259  virtual void removeSubtree(index_t& toRemove) {
260  VBK_TRACE_ZONE_SCOPED;
261  VBK_LOG_DEBUG("remove subtree %s", toRemove.toPrettyString());
262  // save ptr to a previous block
263  auto* prev = toRemove.pprev;
264  VBK_ASSERT(prev && "cannot remove the genesis block");
265 
266  bool isOnMainChain = activeChain_.contains(&toRemove);
267  if (isOnMainChain) {
268  ValidationState dummy;
269  bool success = this->setState(*prev, dummy);
270  VBK_ASSERT_MSG(success, "err: %s", dummy.toString());
271  }
272 
273  // remove this block from 'pnext' set of previous block
274  prev->pnext.erase(&toRemove);
275  forEachNodePostorder<block_t>(
276  toRemove, [&](index_t& next) { removeSingleBlock(next); });
277 
278  // after removal, try to add tip
279  tryAddTip(prev);
280 
281  if (isOnMainChain) {
282  updateTips();
283  }
284  }

316  void invalidateSubtree(index_t& toBeInvalidated,
317  enum BlockValidityStatus reason,
318  bool shouldDetermineBestChain = true) {
319  VBK_TRACE_ZONE_SCOPED;
320  VBK_LOG_DEBUG("Invalidating %s subtree: reason=%d block=%s",
321  block_t::name(),
322  (int)reason,
323  toBeInvalidated.toShortPrettyString());
324 
325  VBK_ASSERT(toBeInvalidated.pprev && "cannot invalidate the genesis block");
326 
327  VBK_ASSERT(isValidInvalidationReason(reason) &&
328  "invalid invalidation reason");
329 
330  if (toBeInvalidated.hasFlags(reason)) {
331  return;
332  }
333 
334  // all descendants of an invalid block are already flagged as
335  // BLOCK_FAILED_CHILD
336  if (!toBeInvalidated.isValid()) {
337  doInvalidate(toBeInvalidated, reason);
338  return;
339  }
340 
341  bool isOnMainChain = activeChain_.contains(&toBeInvalidated);
342  if (isOnMainChain) {
343  ValidationState dummy;
344  bool success = this->setState(*toBeInvalidated.pprev, dummy);
345  VBK_ASSERT(success);
346  }
347 
348  doInvalidate(toBeInvalidated, reason);
349 
350  // flag next subtrees (excluding current block) as BLOCK_FAILED_CHILD
351  for (auto* ptr : toBeInvalidated.pnext) {
352  auto* pnext = ptr;
353  forEachNodePreorder<block_t>(*pnext, [&](index_t& index) {
354  bool valid = index.isValid();
355  doInvalidate(index, BLOCK_FAILED_CHILD);
356  return valid;
357  });
358  }
359 
360  // after invalidation, the previous block might have become a tip
361  tryAddTip(toBeInvalidated.pprev);
362 
363  if (shouldDetermineBestChain) {
364  updateTips();
365  }
366  }

386  /*
387  * `tobeValidated` does NOT necessarily become valid after a call to this
388  * function.
389  */
390  void revalidateSubtree(index_t& toBeValidated,
391  enum BlockValidityStatus reason,
392  bool shouldDetermineBestChain = true) {
393  VBK_TRACE_ZONE_SCOPED;
394  VBK_LOG_DEBUG("Revalidating %s subtree: reason=%d block=%s",
395  block_t::name(),
396  (int)reason,
397  toBeValidated.toShortPrettyString());
398 
399  VBK_ASSERT(toBeValidated.pprev && "cannot revalidate the genesis block");
400  VBK_ASSERT(isValidInvalidationReason(reason) &&
401  "invalid revalidation reason");
402 
403  if (!toBeValidated.hasFlags(reason)) {
404  return;
405  }
406 
407  // if the block has any invalidity flags other than `reason`, its
408  // descendants are already flagged as BLOCK_FAILED_CHILD and should stay so
409  if (toBeValidated.hasFlags(static_cast<enum BlockValidityStatus>(
410  BLOCK_FAILED_MASK & ~reason))) {
411  doReValidate(toBeValidated, reason);
412  return;
413  }
414 
415  doReValidate(toBeValidated, reason);
416 
417  for (auto* pnext : toBeValidated.pnext) {
418  forEachNodePreorder<block_t>(*pnext, [&](index_t& index) -> bool {
419  doReValidate(index, BLOCK_FAILED_CHILD);
420  bool valid = index.isValid();
421  return valid;
422  });
423  }
424 
425  if (shouldDetermineBestChain) {
426  updateTips();
427  }
428  }

564  index_t* doInsertBlockHeader(const std::shared_ptr<block_t>& header,
565  block_height_t bootstrapHeight = 0) {
566  VBK_TRACE_ZONE_SCOPED;
567  VBK_ASSERT(header != nullptr);
568 
569  auto* prev = getBlockIndex(header->getPreviousBlock());
570 
571  VBK_ASSERT_MSG(isBootstrapped() == (prev != nullptr),
572  "already bootstrapped");
573

575  prev == nullptr
576  ? createBootstrapBlockIndex(header->getHash(), bootstrapHeight)
577  : createBlockIndex(header->getHash(), *prev);
578  current->setHeader(std::move(header));
579 
580  return current;

584  index_t* insertBlockHeader(const std::shared_ptr<block_t>& block,
585  block_height_t bootstrapHeight = 0) {
586  VBK_TRACE_ZONE_SCOPED;
587 
588  assertBlockSanity(*block);
589 
590  auto hash = block->getHash();
591  index_t* current = findBlockIndex(hash);
592 
593  if (current != nullptr) {
594  // it is a duplicate
595  if (!current->isDeleted()) {
596  return current;
597  }
598 
599  VBK_ASSERT_MSG(*block == current->getHeader(),
600  "hash collision detected between %s and %s",
601  block->toPrettyString(),
602  current->getHeader().toPrettyString());
603  } else {
604  current = doInsertBlockHeader(block, bootstrapHeight);
605  }
606 
607  current->restore();
608  tryAddTip(current);
609 
610  this->onBlockInserted(current);
611 
612  // raise validity may return false if block is invalid
613  current->raiseValidity(BLOCK_VALID_TREE);
614  return current;
615  }

618  bool loadBlockInner(const stored_index_t& index,
619  bool connectForward,
620  ValidationState& state) {
621  VBK_TRACE_ZONE_SCOPED;
622 
623  VBK_ASSERT(isBootstrapped() && "should be bootstrapped");
624 
625  // quick check if given block is sane
626  auto& root = getRoot();
627  if (connectForward) {
628  VBK_ASSERT_MSG(index.height >= root.getHeight(),
629  format("Blocks can be forward connected after root only. "
630  "index.height: {}, root.height: {}",
631  index.height,
632  root.getHeight()));
633  } else {
634  VBK_ASSERT_MSG(index.height + 1 == root.getHeight(),
635  "Blocks can be backwards connected only to the root");
636  }
637 
638  if (index.height == root.getHeight() &&
639  index.header->getHash() != root.getHash()) {
640  // root is finalized, we can't load a block on same height
641  return state.Invalid("bad-root",
642  format("Can't overwrite root block with block {}",
643  index.toPrettyString()));
644  }
645 
646  auto& header = *index.header;
647  auto currentHash = header.getHash();
648  auto* current = findBlockIndex(currentHash);
649 
650  // we can not load a block, which already exists on chain and is not a
651  // bootstrap block
652  if (current && !current->isDeleted() &&
653  !current->hasFlags(BLOCK_BOOTSTRAP)) {
654  return state.Invalid(
655  "block-exists",
656  "Found duplicate block, which is not bootstrap block");
657  }
658 
659  if (current) {
660  if (current->isDeleted()) {
661  current->restore();
662  }
663  } else {
664  auto* prev = getBlockIndex(header.getPreviousBlock());
665  // we can not load block which is not connected to the
666  // bootstrap block
667  if (prev == nullptr) {
668  if (connectForward) {
669  return state.Invalid("bad-prev",
670  "Block does not connect to current tree");
671  }
672 
673  if (root.getHeader().getPreviousBlock() != makePrevHash(currentHash)) {
674  return state.Invalid("bad-block-hash",
675  format("Can't replace root block with block {}",
676  index.toPrettyString()));
677  }
678 
679  current = createBootstrapBlockIndex(currentHash, index.height);
680  current->restore();
681  current->pnext.insert(&root);
682  VBK_ASSERT(root.pprev == nullptr);
683  root.pprev = current;
684 
685  } else {
686  current = createBlockIndex(currentHash, *prev);
687  current->restore();
688  }
689  }
690 
691  VBK_ASSERT(current);
692  VBK_ASSERT(!current->isDeleted());
693 
694  // touchBlockIndex may return an existing block (one of bootstrap blocks),
695  // so backup its 'pnext'
696  auto next = current->pnext;
697  auto prev = current->pprev;
698  // copy all fields
699  current->mergeFrom(index);
700  // FIXME: this is an ugly HACK
701  // clear inmem fields
702  current->setNullInmemFields();
703  // recover pnext
704  current->pnext = next;
705  // recover pprev
706  current->pprev = prev;
707 
708  // check for valid previous block
709  if (current->pprev != nullptr) {
710  // prev block found
711  auto expectedHeight = current->pprev->getHeight() + 1;
712  if (current->getHeight() != expectedHeight) {
713  return state.Invalid("bad-height");
714  }
715  }
716 
717  // set a new bootstrap block which is the prev block for the current
718  // bootstrap block
719  if ((current->pprev == nullptr) && (activeChain_.first() != current) &&
720  !connectForward) {
721  activeChain_.prependRoot(current);
722  }
723 
724  VBK_ASSERT(!current->isDeleted());
725  current->raiseValidity(BLOCK_VALID_TREE);
726  current->unsetDirty();
727 
728  tryAddTip(current);
729 
730  return true;
731  }

733  std::string toPrettyString(size_t level = 0) const {
734  auto tip = activeChain_.tip();
735  std::string pad(level, ' ');
736 
737  std::string blocksStr{};
738  // sort blocks by height
739  std::vector<std::pair<int, index_t*>> byheight;
740  byheight.reserve(blocks_.size());
741  for (const auto& p : blocks_) {
742  // FIXME: ugly hack due to trees being compared via toPrettyString in
743  // tests
744  if (!p.second->isDeleted()) {
745  byheight.emplace_back(p.second->getHeight(), p.second.get());
746  }
747  }
748  std::sort(byheight.rbegin(), byheight.rend());
749  for (const auto& p : byheight) {
750  blocksStr += (p.second->toPrettyString(level + 2) + "\n");
751  }
752 
753  std::string tipsStr{};
754  for (const auto* _tip : tips_) {
755  tipsStr += (_tip->toPrettyString(level + 2) + "\n");
756  }
757 
758  return format("{}{{tip={}}}\n{}{{blocks=\n{}{}}}\n{}{{tips=\n{}{}}}",
759  pad,
760  (tip ? tip->toPrettyString() : "<empty>"),
761  pad,
762  blocksStr,
763  pad,
764  pad,
765  tipsStr,
766  pad);
767  }

846  virtual bool finalizeBlockImpl(index_t& index,
847  // see config.preserveBlocksBehindFinal()
848  int32_t preserveBlocksBehindFinal,
849  ValidationState& state) {
850  VBK_TRACE_ZONE_SCOPED;
851  VBK_LOG_DEBUG("Finalize %s, preserve %d blocks behind",
852  index.toShortPrettyString(),
853  preserveBlocksBehindFinal);
854 
855  index_t* finalizedBlock = &index;
856 
857  // prereq is not met - finalized block must be on active chain
858  if (!activeChain_.contains(finalizedBlock)) {
859  return state.Invalid("block-not-on-active-chain");
860  }
861 
862  // we need to clarify which block will be final. we can not remove blocks
863  // which has not been saved into the storage
864  for (auto* walkBlock = finalizedBlock; walkBlock != nullptr;
865  walkBlock = walkBlock->pprev) {
866  if (walkBlock->isDirty()) {
867  finalizedBlock = walkBlock;
868  }
869  }
870 
871  // first, erase candidates from tips_ that will never be activated
872  erase_if<decltype(tips_), index_t*>(
873  tips_, [this, &finalizedBlock](index_t* tip) -> bool {
874  VBK_ASSERT(tip);
875 
876  // tip from active chain can not be outdated
877  if (!activeChain_.contains(tip) &&
878  isBlockOutdated(*finalizedBlock, *tip)) {
879  // we need to clarify which block will be final. we can not remove
880  // blocks
881  // which has not been saved into the storage
882  bool newFinalized = false;
883  auto* walkBlock = tip;
884  for (; tip != nullptr && !activeChain_.contains(walkBlock);
885  walkBlock = walkBlock->pprev) {
886  if (walkBlock->isDirty()) {
887  newFinalized = true;
888  }
889  }
890 
891  if (newFinalized) {
892  finalizedBlock = walkBlock;
893  }
894 
895  return true;
896  }
897 
898  return false;
899  });
900 
901  // second, update active chain (it should start with
902  // 'finalizedBlock' but we also need to preserve `preserveBlocksBehindFinal`
903  // blocks before it). all outdated blocks behind `finalizedBlock` block will
904  // be deallocated
905  int32_t firstBlockHeight =
906  finalizedBlock->getHeight() - preserveBlocksBehindFinal;
907  int32_t bootstrapBlockHeight = getRoot().getHeight();
908  firstBlockHeight = std::max(bootstrapBlockHeight, firstBlockHeight);
909 
910  // before we deallocate subtree, disconnect "new root block" from previous
911  // tree
912  auto* newRoot = activeChain_[firstBlockHeight];
913  VBK_ASSERT(newRoot);
914  auto* rootPrev = newRoot->pprev;
915  if (newRoot->pprev != nullptr) {
916  newRoot->pprev->pnext.erase(newRoot);
917  newRoot->pprev = nullptr;
918 
919  // do deallocate
920  deallocateTree(*rootPrev);
921 
922  // erase "parallel" blocks - blocks that are on same height as `index`,
923  // but since `index` is final, will never be active.
924  if (finalizedBlock->pprev != nullptr) {
925  auto parallelBlocks = finalizedBlock->pprev->pnext;
926  parallelBlocks.erase(finalizedBlock);
927  for (auto* par : parallelBlocks) {
928  // disconnect `par` from prev block
929  if (par->pprev != nullptr) {
930  par->pprev->pnext.erase(par);
931  par->pprev = nullptr;
932  }
933  deallocateTree(*par);
934  }
935  }
936  }
937 
938  // update active chain
939  VBK_ASSERT(firstBlockHeight >= bootstrapBlockHeight);
940  size_t deallocatedBlocks = firstBlockHeight - bootstrapBlockHeight;
941  decreaseAppliedBlockCount(deallocatedBlocks);
942  activeChain_ = Chain<index_t>(firstBlockHeight, activeChain_.tip());
943 
944  // fourth, mark `index` and all predecessors as finalized
945  index_t* ptr = finalizedBlock;
946  while (ptr != nullptr && !ptr->finalized) {
947  ptr->finalized = true;
948  ptr = ptr->pprev;
949  }
950 
951  return true;
952  }
