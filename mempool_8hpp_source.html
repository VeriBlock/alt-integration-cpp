99  template <typename T,
100  typename = typename std::enable_if<IsPopPayload<T>::value>::type>
101  bool isKnown(const typename T::id_t& id) const {
102  // is `id` in mempool?
103  auto* inmempool = get<T>(id);
104  if (inmempool != nullptr) {
105  return true;
106  }
107 
108  // is `id` in blockchain?
109  auto& pl = getPayloadsIndex();
110  std::vector<uint8_t> v(id.begin(), id.end());
111  const auto& containing = pl.getContainingAltBlocks(v);
112  if (containing.empty()) {
113  return false;
114  }
115 
116  // check if any of containing is on active chain
117  auto& tree = mempool_tree_.alt();
118  for (const auto& c : containing) {
119  auto* index = tree.getBlockIndex(c);
120  if (index == nullptr) {
121  // block is deleted (too old)
122  continue;
123  }
124  if (tree.getBestChain().contains(index)) {
125  return true;
126  }
127  }
128 
129  // `id` is not on active chain
130  return false;
131  }

134  template <typename T,
135  typename = typename std::enable_if<IsPopPayload<T>::value>::type>
136  const T* get(const typename T::id_t& id) const {
137  const auto& map = getMap<T>();
138  auto it = map.find(id);
139  if (it != map.end()) {
140  return it->second.get();
141  }
142 
143  const auto& inflight = getInFlightMap<T>();
144  auto it2 = inflight.find(id);
145  if (it2 != inflight.end()) {
146  return it2->second.get();
147  }
148 
149  return nullptr;
150  }

439 template <typename Value, typename T>
440 inline void mapToJson(Value& obj,
441  const MemPool& mp,
442  const std::string& key,
443  bool verbose = false) {
444  auto arr = json::makeEmptyArray<Value>();
445  for (auto& p : mp.getMap<T>()) {
446  if (verbose) {
447  json::arrayPushBack(arr, ToJSON<Value>(*p.second));
448  } else {
449  json::arrayPushBack(arr, ToJSON<Value>(p.first));
450  }
451  }
452  for (auto& p : mp.getInFlightMap<T>()) {
453  if (verbose) {
454  json::arrayPushBack(arr, ToJSON<Value>(*p.second));
455  } else {
456  json::arrayPushBack(arr, ToJSON<Value>(p.first));
457  }
458  }
459  json::putKV(obj, key, arr);
460 }
461 } // namespace detail
