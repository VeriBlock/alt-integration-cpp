97  template <typename T,
98  typename = typename std::enable_if<IsPopPayload<T>::value>::type>
99  bool isKnown(const typename T::id_t& id) const {
100  // is `id` in mempool?
101  auto* inmempool = get<T>(id);
102  if (inmempool != nullptr) {
103  return true;
104  }
105 
106  // is `id` in blockchain?
107  auto& pl = getPayloadsIndex();
108  std::vector<uint8_t> v(id.begin(), id.end());
109  const auto& containing = pl.getContainingAltBlocks(v);
110  return !containing.empty();
111  }

114  template <typename T,
115  typename = typename std::enable_if<IsPopPayload<T>::value>::type>
116  const T* get(const typename T::id_t& id) const {
117  const auto& map = getMap<T>();
118  auto it = map.find(id);
119  if (it != map.end()) {
120  return it->second.get();
121  }
122 
123  const auto& inflight = getInFlightMap<T>();
124  auto it2 = inflight.find(id);
125  if (it2 != inflight.end()) {
126  return it2->second.get();
127  }
128 
129  return nullptr;
130  }

314  template <typename T>
315  void makePayloadConnected(const std::shared_ptr<T>& t) {
316  auto& signal = getSignal<T>();
317  auto& inflight = getInFlightMapMut<T>();
318  auto& connected = getMapMut<T>();
319 
320  auto id = t->getId();
321  connected[id] = t;
322  inflight.erase(id);
323  signal.emit(*t);

326  template <typename POP>
327  void cleanupStale(std::vector<std::shared_ptr<POP>>& c,
328  std::function<void(POP&)> remove) {
329  for (auto it = c.begin(); it != c.end();) {
330  auto& pl = **it;
331  ValidationState state;
332  auto valid = mempool_tree_.checkContextually(pl, state);
333  if (!valid) {
334  remove(pl);
335  it = c.erase(it);
336  } else {
337  ++it;
338  }
339  }
340  }

342  template <typename POP>
343  void cleanupStale(payload_value_sorted_map<POP>& c) {
344  for (auto it = c.begin(); it != c.end();) {
345  auto& pl = *it->second;
346  ValidationState state;
347  auto valid = mempool_tree_.checkContextually(pl, state);
348  it = !valid ? c.erase(it) : std::next(it);
349  }
350  }

401 inline void mapToJson(Value& obj, const MemPool& mp, const std::string& key) {
402  auto arr = json::makeEmptyArray<Value>();
403  for (auto& p : mp.getMap<T>()) {
