100 template <typename T,
101 typename = typename std::enable_if<IsPopPayload<T>::value>::type>
102 bool isKnown(const typename T::id_t& id, bool onlyInMempool = false) const {
103 // is `id` in mempool?
104 auto* inmempool = get<T>(id);
105 if (inmempool != nullptr) {
106 return true;
107 }
108
109 if (onlyInMempool) {
110 // we did not find in mempool, so exit early
111 return false;
112 }
113
114 std::vector<uint8_t> v(id.begin(), id.end());
115
116 auto& tree = mempool_tree_.alt();
117 auto& fpl = tree.getFinalizedPayloadsIndex();
118 const auto* blockhash = fpl.find(v);
119 if (blockhash != nullptr) {
120 // finalized payloads index stores only payloads from finalized blocks.
121 // all finalized blocks are on active chain.
122 return true;
123 }
124
125 auto& pl = tree.getPayloadsIndex();
126 // is `id` in payloads index?
127 const auto& set = pl.find(v);
128 if (set.empty()) {
129 return false;
130 }
131
132 // check if any of candidates is on active chain
133 for(const auto& hash: set) {
134 const auto* candidate = tree.getBlockIndex(hash);
135 // all candidates must exist in a tree after split on PayloadsIndex+FinalizedPayloadsIndex.
136 // only finalized blocks may not exist in a tree.
137 VBK_ASSERT_MSG(candidate != nullptr, candidate->toPrettyString());
138 if(tree.getBestChain().contains(candidate)) {
139 // candidate is on main chain
140 return true;
141 }
142 }
143
144 // none of candidates are on main chain
145 return false;
146 }

149 template <typename T,
150 typename = typename std::enable_if<IsPopPayload<T>::value>::type>
151 const T* get(const typename T::id_t& id) const {
152 const auto& map = getMap<T>();
153 auto it = map.find(id);
154 if (it != map.end()) {
155 return it->second.get();
156 }
157
158 const auto& inflight = getInFlightMap<T>();
159 auto it2 = inflight.find(id);
160 if (it2 != inflight.end()) {
161 return it2->second.get();
162 }
163
164 return nullptr;
165 }

184 template <typename T,
185 typename = typename std::enable_if<IsPopPayload<T>::value>::type>
186 SubmitResult submit(Slice<const uint8_t> bytes, ValidationState& state) {
187 ReadStream stream(bytes);
188 T payload;
189 if (!DeserializeFromVbkEncoding(stream, payload, state)) {
190 return {FAILED_STATELESS,
191 state.Invalid("pop-mempool-submit-deserialize")};
192 }
193
194 return submit<T>(payload, state);

357 template <typename T>
358 void makePayloadConnected(const std::shared_ptr<T>& t) {
359 auto& signal = getSignal<T>();
360 auto& inflight = getInFlightMapMut<T>();
361 auto& connected = getMapMut<T>();
362
363 auto id = t->getId();
364 connected[id] = t;
365 inflight.erase(id);
366 signal.emit(*t);
367 }

369 template <typename POP>
370 void cleanupStale(std::set<std::shared_ptr<POP>,
371 VbkPayloadsRelations::AtvCombinedComparator>& c,
372 std::function<void(POP&)> remove) {
373 for (auto it = c.begin(); it != c.end();) {
374 auto& pl = **it;
375 ValidationState state;
376 auto valid = mempool_tree_.checkContextually(pl, state);
377 it = !valid ? (remove(pl), c.erase(it)) : std::next(it);
378 }
379 }

450template <typename Value, typename T>
451inline void mapToJson(Value& obj,
452 const MemPool& mp,
453 const std::string& key,
454 bool verbose = false) {
455 auto arr = json::makeEmptyArray<Value>();
456 for (auto& p : mp.getMap<T>()) {
457 if (verbose) {
458 json::arrayPushBack(arr, ToJSON<Value>(*p.second));
459 } else {
460 json::arrayPushBack(arr, ToJSON<Value>(p.first));
461 }
462 }
463 for (auto& p : mp.getInFlightMap<T>()) {
464 if (verbose) {
465 json::arrayPushBack(arr, ToJSON<Value>(*p.second));
466 } else {
467 json::arrayPushBack(arr, ToJSON<Value>(p.first));
468 }
469 }
470 json::putKV(obj, key, arr);
471}
472} // namespace detail
473
475template <typename Value>
476Value ToJSON(const MemPool& mp, bool verbose = false) {
477 auto obj = json::makeEmptyObject<Value>();
478
479 detail::mapToJson<Value, VbkBlock>(obj, mp, "vbkblocks", verbose);
480 detail::mapToJson<Value, ATV>(obj, mp, "atvs", verbose);
481 detail::mapToJson<Value, VTB>(obj, mp, "vtbs", verbose);
482
483 return obj;
484}
