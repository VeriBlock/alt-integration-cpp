257 template <size_t N>
258 struct std::hash<altintegration::Blob<N>> {
259  size_t operator()(std::true_type, const altintegration::Blob<N>& x) const {
260  return x.getLow64();
261  }
262 
263  size_t operator()(std::false_type, const altintegration::Blob<N>& x) const {
264  return std::hash<std::string>{}(std::string(x.begin(), x.end()));
265  }
266 
267  size_t operator()(const altintegration::Blob<N>& x) const {
268  // if N >= 8, use getLow64 impl
269  // else use std::string hasher
270  typename std::conditional<N >= 8, std::true_type, std::false_type>::type f;
271  return operator()(f, x);
272  }
273 };
