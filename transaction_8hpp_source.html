105 template <typename Stream>
106 inline void UnserializeTransaction(Transaction& tx, Stream& s) {
107  // TODO: fix
108  // const bool fAllowWitness = !(s.GetVersion() &
109  // SERIALIZE_TRANSACTION_NO_WITNESS);
110  const bool fAllowWitness = false;
111 
112  UnserializeBtc(s, tx.nVersion);
113  unsigned char flags = 0;
114  tx.vin.clear();
115  tx.vout.clear();
116  /* Try to read the vin. In case the dummy is there, this will be read as an
117  * empty vector. */
118  UnserializeBtc(s, tx.vin);
119  if (tx.vin.size() == 0 && fAllowWitness) {
120  /* We read a dummy or an empty vin. */
121  UnserializeBtc(s, flags);
122  if (flags != 0) {
123  UnserializeBtc(s, tx.vin);
124  UnserializeBtc(s, tx.vout);
125  }
126  } else {
127  /* We read a non-empty vin. Assume a normal vout follows. */
128  UnserializeBtc(s, tx.vout);
129  }
130  if ((flags & 1) && fAllowWitness) {
131  /* The witness flag is present, and we support witnesses. */
132  flags ^= 1;
133  for (size_t i = 0; i < tx.vin.size(); i++) {
134  UnserializeBtc(s, tx.vin[i].scriptWitness);
135  }
136  if (!tx.HasWitness()) {
137  /* It's illegal to encode witnesses when all witness stacks are empty. */
138  throw std::ios_base::failure("Superfluous witness record");
139  }
140  }
141  if (flags) {
142  /* Unknown flag in the serialization */
143  throw std::ios_base::failure("Unknown transaction optional data");
144  }
145  UnserializeBtc(s, tx.nLockTime);
146 }

148 template <typename Stream>
149 inline void SerializeTransaction(const Transaction& tx, Stream& s) {
150  // TODO: fix
151  // const bool fAllowWitness = !(s.GetVersion() &
152  // SERIALIZE_TRANSACTION_NO_WITNESS);
153  const bool fAllowWitness = false;
154 
155  SerializeBtc(s, tx.nVersion);
156  unsigned char flags = 0;
157  // Consistency check
158  if (fAllowWitness) {
159  /* Check whether witnesses need to be serialized. */
160  if (tx.HasWitness()) {
161  flags |= 1;
162  }
163  }
164  if (flags) {
165  /* Use extended format in case witnesses are to be serialized. */
166  std::vector<TxIn> vinDummy;
167  SerializeBtc(s, vinDummy);
168  SerializeBtc(s, flags);
169  }
170  SerializeBtc(s, tx.vin);
171  SerializeBtc(s, tx.vout);
172  if (flags & 1) {
173  for (size_t i = 0; i < tx.vin.size(); i++) {
174  SerializeBtc(s, tx.vin[i].scriptWitness);
175  }
176  }
177  SerializeBtc(s, tx.nLockTime);
178 }
