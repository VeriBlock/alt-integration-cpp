16 template <typename StableBlockTree>
17 struct TempBlockTree {
19  template <typename BlockIndexT>
20  struct TempBlockIndex : public BlockIndexT {
21  using base = BlockIndexT;
22  using base::base;
23 
24  ~TempBlockIndex() {
25  // if alttree is destroyed before temp block tree, then
26  // `pprev` may point to deallocated block. to solve this, override
27  // destructor of temp block index, as we don't care in which order temp
28  // blocks are destroyed.
29  this->pprev = nullptr;
30  this->pnext.clear();
31  }
32  };

34  using block_tree_t = StableBlockTree;
35  using block_t = typename block_tree_t::block_t;
36  using index_t = typename block_tree_t::index_t;
37  using wrapped_index_t = TempBlockIndex<index_t>;
38  using prev_block_hash_t = typename StableBlockTree::prev_block_hash_t;
39  using block_index_t =

54  template <typename T,
55  typename = typename std::enable_if<
56  std::is_same<T, typename block_t::hash_t>::value ||
57  std::is_same<T, typename block_t::prev_hash_t>::value>::type>
58  const index_t* getTempBlockIndex(const T& hash) const {
59  auto shortHash = tree_->makePrevHash(hash);
60  auto it = temp_blocks_.find(shortHash);
61  return it == temp_blocks_.end() ? nullptr : it->second.get();
62  }
63 
65  template <typename T,
66  typename = typename std::enable_if<
67  std::is_same<T, typename block_t::hash_t>::value ||
68  std::is_same<T, typename block_t::prev_hash_t>::value>::type>
69  index_t* getTempBlockIndex(const T& hash) {
70  const auto& t = as_const(*this);
71  return const_cast<index_t*>(t.getTempBlockIndex(hash));
72  }
73 
74  template <typename T,
75  typename = typename std::enable_if<
76  std::is_same<T, typename block_t::hash_t>::value ||
77  std::is_same<T, typename block_t::prev_hash_t>::value>::type>
78  const index_t* getBlockIndex(const T& hash) const {
79  auto* index = getTempBlockIndex(hash);
80  return index == nullptr ? tree_->getBlockIndex(hash) : index;
81  }

84  template <typename T,
85  typename = typename std::enable_if<
86  std::is_same<T, typename block_t::hash_t>::value ||

97  bool acceptBlockHeader(std::shared_ptr<block_t> header,
98  ValidationState& state) {
99  VBK_ASSERT(header);
100  auto* prev = getBlockIndex(header->getPreviousBlock());
101  if (prev == nullptr) {
102  return state.Invalid(
103  block_t::name() + "-bad-prev-block",
104  "can not find previous block: " + HexStr(header->getPreviousBlock()));
105  }
106 
107  auto index = insertBlockHeader(std::move(header), prev);
108  VBK_ASSERT(index != nullptr &&
109  "insertBlockHeader should have never returned nullptr");
110 
111  return true;
112  }
113 
114  const block_tree_t& getStableTree() const { return *tree_; }

119  void cleanUpStaleBlocks() {
120  for (auto it = temp_blocks_.begin(); it != temp_blocks_.end();) {
121  auto short_hash = tree_->makePrevHash(it->second->getHash());
122  auto* index = tree_->getBlockIndex(short_hash);
123  it = index ? temp_blocks_.erase(it) : std::next(it);
124  }
125  }
126 
127  void clear() { temp_blocks_.clear(); }
128 
129  private:
130  index_t* insertBlockHeader(std::shared_ptr<block_t> header, index_t* prev) {
131  auto hash = header->getHash();
132  index_t* current = getBlockIndex(hash);
133  if (current != nullptr) {
134  // it is a duplicate
135  return current;
136  }
137 
138  current = doInsertBlockHeader(std::move(header), prev);
139  VBK_ASSERT(current != nullptr);
140 
141  return current;
142  }
143 
144  index_t* doInsertBlockHeader(std::shared_ptr<block_t> header, index_t* prev) {
145  VBK_ASSERT(header != nullptr);
146  index_t* current = touchBlockIndex(header->getHash());
147  current->setHeader(std::move(header));
148  current->pprev = prev;
149 
150  if (current->isRoot()) {
151  current->setHeight(0);
152  } else {
153  current->setHeight(current->pprev->getHeight() + 1);
154  }
155 
156  return current;
157  }
158 
159  index_t* touchBlockIndex(const typename block_t::hash_t& hash) {
160  auto shortHash = tree_->makePrevHash(hash);
161  auto it = temp_blocks_.find(shortHash);
162  if (it != temp_blocks_.end()) {
163  return it->second.get();
164  }
165 
166  // intentionally do not pass 'prev' here - we don't want 'stable' blocks
167  // have pnext ptr to temp blocks.
168  auto newIndex = make_unique<wrapped_index_t>(0);
169  newIndex->setNull();
170  it = temp_blocks_.emplace(shortHash, std::move(newIndex)).first;
171  return it->second.get();
172  }
173 
174  private:
175  block_index_t temp_blocks_;
176  const block_tree_t* tree_;
177 };
