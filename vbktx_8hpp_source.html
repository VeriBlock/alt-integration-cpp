93template <typename JsonValue>
94JsonValue ToJSON(const VbkTx& tx) {
95 JsonValue obj = json::makeEmptyObject<JsonValue>();
96 tx.networkOrType.networkType.template putJson<JsonValue>(obj);
97 json::putStringKV(obj, "hash", tx.getHash().toHex());
98 json::putIntKV(obj, "type", tx.networkOrType.typeId);
99 json::putStringKV(obj, "sourceAddress", tx.sourceAddress.toString());
100 json::putIntKV(obj, "sourceAmount", tx.sourceAmount.units);
101 json::putArrayKV(obj, "outputs", tx.outputs);
102 json::putIntKV(obj, "signatureIndex", tx.signatureIndex);
103 json::putKV(obj, "publicationData", ToJSON<JsonValue>(tx.publicationData));
104 json::putStringKV(obj, "signature", HexStr(tx.signature));
105 json::putStringKV(obj, "publicKey", HexStr(tx.publicKey));
106 return obj;
107}
