84template <typename JsonValue>
85JsonValue ToJSON(const VbkTx& tx) {
86 JsonValue obj = json::makeEmptyObject<JsonValue>();
87 tx.networkOrType.networkType.template putJson<JsonValue>(obj);
88 json::putStringKV(obj, "hash", tx.getHash().toHex());
89 json::putIntKV(obj, "type", tx.networkOrType.typeId);
90 json::putStringKV(obj, "sourceAddress", tx.sourceAddress.toString());
91 json::putIntKV(obj, "sourceAmount", tx.sourceAmount.units);
92 json::putArrayKV(obj, "outputs", tx.outputs);
93 json::putIntKV(obj, "signatureIndex", tx.signatureIndex);
94 json::putKV(obj, "publicationData", ToJSON<JsonValue>(tx.publicationData));
95 json::putStringKV(obj, "signature", HexStr(tx.signature));
96 json::putStringKV(obj, "publicKey", HexStr(tx.publicKey));
97 return obj;
98}
