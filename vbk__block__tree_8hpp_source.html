265 template <typename JsonValue>
266 JsonValue ToJSON(const BlockIndex<VbkBlock>& i) {
267  auto obj = json::makeEmptyObject<JsonValue>();
268  json::putStringKV(obj, "chainWork", i.chainWork.toHex());
269 
270  std::vector<uint256> endorsements;
271  for (const auto& e : i.getContainingEndorsements()) {
272  endorsements.push_back(e.first);
273  }
274  json::putArrayKV(obj, "containingEndorsements", endorsements);
275 
276  std::vector<uint256> endorsedBy;
277  for (const auto* e : i.getEndorsedBy()) {
278  endorsedBy.push_back(e->id);
279  }
280  json::putArrayKV(obj, "endorsedBy", endorsedBy);
281  json::putIntKV(obj, "height", i.getHeight());
282  json::putKV(obj, "header", ToJSON<JsonValue>(i.getHeader()));
283  json::putIntKV(obj, "status", i.getStatus());
284  json::putIntKV(obj, "altrefs", i.refCount());
285 
286  auto stored = json::makeEmptyObject<JsonValue>();
287  json::putArrayKV(stored, "vtbids", i.getPayloadIds<VTB>());
288  json::putKV(obj, "stored", stored);
289 
290  auto bopEndorsements = json::makeEmptyArray<JsonValue>();
291  for (const auto* e : i.getBlockOfProofEndorsement()) {
292  if (e == nullptr) {
293  continue;
294  }
295  json::arrayPushBack(bopEndorsements, ToJSON<JsonValue>(e->getId()));
296  }
297  json::putKV(obj, "blockOfProofEndorsements", bopEndorsements);
298 
299  return obj;
300 }
