240template <typename JsonValue>
241JsonValue ToJSON(const BlockIndex<VbkBlock>& i) {
242 auto obj = json::makeEmptyObject<JsonValue>();
243 json::putStringKV(obj, "chainWork", i.chainWork.toHex());
244
245 std::vector<uint256> endorsements;
246 for (const auto& e : i.getContainingEndorsements()) {
247 endorsements.push_back(e.first);
248 }
249 json::putArrayKV(obj, "containingEndorsements", endorsements);
250
251 std::vector<uint256> endorsedBy;
252 for (const auto* e : i.getEndorsedBy()) {
253 endorsedBy.push_back(e->id);
254 }
255 json::putArrayKV(obj, "endorsedBy", endorsedBy);
256 json::putIntKV(obj, "height", i.getHeight());
257 json::putKV(obj, "header", ToJSON<JsonValue>(i.getHeader()));
258 json::putIntKV(obj, "status", i.getStatus());
259 json::putIntKV(obj, "altrefs", i.refCount());
260
261 auto stored = json::makeEmptyObject<JsonValue>();
262 json::putArrayKV(stored, "vtbids", i.getPayloadIds<VTB>());
263 json::putKV(obj, "stored", stored);
264
265 auto bopEndorsements = json::makeEmptyArray<JsonValue>();
266 for (const auto* e : i.getBlockOfProofEndorsement()) {
267 if (e == nullptr) {
268 continue;
269 }
270 json::arrayPushBack(bopEndorsements, ToJSON<JsonValue>(e->getId()));
271 }
272 json::putKV(obj, "blockOfProofEndorsements", bopEndorsements);
273
274 return obj;
275}

293 }
294 json::putKV(obj, "blockOfProofEndorsements", bopEndorsements);
295
