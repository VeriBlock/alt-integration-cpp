242 template <typename JsonValue>
243 JsonValue ToJSON(const BlockIndex<VbkBlock>& i) {
244  auto obj = json::makeEmptyObject<JsonValue>();
245  json::putStringKV(obj, "chainWork", i.chainWork.toHex());
246 
247  std::vector<uint256> endorsements;
248  for (const auto& e : i.getContainingEndorsements()) {
249  endorsements.push_back(e.first);
250  }
251  json::putArrayKV(obj, "containingEndorsements", endorsements);
252 
253  std::vector<uint256> endorsedBy;
254  for (const auto* e : i.getEndorsedBy()) {
255  endorsedBy.push_back(e->id);
256  }
257  json::putArrayKV(obj, "endorsedBy", endorsedBy);
258  json::putIntKV(obj, "height", i.getHeight());
259  json::putKV(obj, "header", ToJSON<JsonValue>(i.getHeader()));
260  json::putIntKV(obj, "status", i.getStatus());
261  json::putIntKV(obj, "altrefs", i.refCount());
262 
263  auto stored = json::makeEmptyObject<JsonValue>();
264  json::putArrayKV(stored, "vtbids", i.getPayloadIds<VTB>());
265  json::putKV(obj, "stored", stored);
266 
267  auto bopEndorsements = json::makeEmptyArray<JsonValue>();
268  for (const auto* e : i.getBlockOfProofEndorsement()) {
269  if (e == nullptr) {
270  continue;
271  }
272  json::arrayPushBack(bopEndorsements, ToJSON<JsonValue>(e->getId()));
273  }
274  json::putKV(obj, "blockOfProofEndorsements", bopEndorsements);
275 
276  return obj;
277 }
