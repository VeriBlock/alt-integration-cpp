252 template <typename JsonValue>
253 JsonValue ToJSON(const BlockIndex<VbkBlock>& i) {
254  auto obj = json::makeEmptyObject<JsonValue>();
255  json::putStringKV(obj, "chainWork", i.chainWork.toHex());
256 
257  std::vector<uint256> endorsements;
258  for (const auto& e : i.getContainingEndorsements()) {
259  endorsements.push_back(e.first);
260  }
261  json::putArrayKV(obj, "containingEndorsements", endorsements);
262 
263  std::vector<uint256> endorsedBy;
264  for (const auto* e : i.getEndorsedBy()) {
265  endorsedBy.push_back(e->id);
266  }
267  json::putArrayKV(obj, "endorsedBy", endorsedBy);
268  json::putIntKV(obj, "height", i.getHeight());
269  json::putKV(obj, "header", ToJSON<JsonValue>(i.getHeader()));
270  json::putIntKV(obj, "status", i.getStatus());
271  json::putIntKV(obj, "altrefs", i.refCount());
272 
273  auto stored = json::makeEmptyObject<JsonValue>();
274  json::putArrayKV(stored, "vtbids", i.getPayloadIds<VTB>());
275  json::putKV(obj, "stored", stored);
276 
277  auto bopEndorsements = json::makeEmptyArray<JsonValue>();
278  for (const auto* e : i.getBlockOfProofEndorsement()) {
279  if (e == nullptr) {
280  continue;
281  }
282  json::arrayPushBack(bopEndorsements, ToJSON<JsonValue>(e->getId()));
283  }
284  json::putKV(obj, "blockOfProofEndorsements", bopEndorsements);
285 
286  return obj;
287 }
