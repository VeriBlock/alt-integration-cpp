230template <typename JsonValue>
231JsonValue ToJSON(const BlockIndex<VbkBlock>& i) {
232 auto obj = json::makeEmptyObject<JsonValue>();
233 json::putStringKV(obj, "chainWork", i.chainWork.toHex());
234
235 std::vector<uint256> endorsements;
236 for (const auto& e : i.getContainingEndorsements()) {
237 endorsements.push_back(e.first);
238 }
239 json::putArrayKV(obj, "containingEndorsements", endorsements);
240
241 std::vector<uint256> endorsedBy;
242 for (const auto* e : i.getEndorsedBy()) {
243 endorsedBy.push_back(e->id);
244 }
245 json::putArrayKV(obj, "endorsedBy", endorsedBy);
246 json::putIntKV(obj, "height", i.getHeight());
247 json::putKV(obj, "header", ToJSON<JsonValue>(i.getHeader()));
248 json::putIntKV(obj, "status", i.getStatus());
249 json::putIntKV(obj, "altrefs", i.refCount());
250
251 auto stored = json::makeEmptyObject<JsonValue>();
252 json::putArrayKV(stored, "vtbids", i.getPayloadIds<VTB>());
253 json::putKV(obj, "stored", stored);
254
255 auto bopEndorsements = json::makeEmptyArray<JsonValue>();
256 for (const auto* e : i.getBlockOfProofEndorsement()) {
257 if (e == nullptr) {
258 continue;
259 }
260 json::arrayPushBack(bopEndorsements, ToJSON<JsonValue>(e->getId()));
261 }
262 json::putKV(obj, "blockOfProofEndorsements", bopEndorsements);
263
264 return obj;
265}

278 for (const auto* e : i.getBlockOfProofEndorsement()) {

282 json::arrayPushBack(bopEndorsements, ToJSON<JsonValue>(e->getId()));
283 }
284 json::putKV(obj, "blockOfProofEndorsements", bopEndorsements);
285
286 return obj;
