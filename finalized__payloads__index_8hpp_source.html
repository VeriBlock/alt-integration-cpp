19struct FinalizedPayloadsIndex {
20 using hash_t = typename ContainerIndex::hash_t;
21 using payload_id = std::vector<uint8_t>;
22
23 const hash_t* find(const payload_id& pid) const {
24 auto it = map_.find(pid);
25 if (it == map_.end()) {
26 return nullptr;
27 }
28
29 return &it->second;
30 }
31
32 bool empty() const { return map_.empty(); }
33
34 size_t size() const { return map_.size(); }
35
36 void add(const payload_id& pid, const hash_t& hash) {
37 auto it = map_.find(pid);
38 VBK_ASSERT(it == map_.end());
39 map_.insert(std::make_pair(pid, hash));
40 }
41
42 void remove(const payload_id& pid) {
43 auto it = map_.find(pid);
44 VBK_ASSERT(it != map_.end());
45 map_.erase(pid);
46 }
47
48 private:
49 std::unordered_map<payload_id, hash_t> map_;
50};
